        \quad $\to$ $\mathcal{F}\{f\}=A_k=\frac{1}{2\pi}\int_{-\pi}^{\pi} f(x) e^{-ikx} dx$ \quad \scalebox{0.9}{(i.e. Fourier Transform)} \qquad \ok{Remark}: For real func: $A_{-k}=\overline{A_k}$

        \quad \imp{Property}: $\frac{df}{dx}=\sum_{n=-\infty}^{\infty} (ik) A_k e^{ikx}$ \quad $\Rightarrow$ \quad $f^{(m)}(x)=\sum_{n=-\infty}^{\infty} (ik)^m A_k e^{ikx}$
    \end{MyTh}

    \begin{MyTh}{Discrete Fourier Transform (DFT)} \tiny
        Assume $x_{n}$ are \ok{equally-spaced} points with $n=0,1,...,N-1$

        \vspace{-1pt}
        \begin{enumerate}
            \item \ok{Fourier Transform}: $\mathcal{F}\{f\}\approx\widehat{f}_k= \frac{1}{N}\sum_{n=0}^{N-1} f(x_n) e^{-2\pi i k n / N}$
            \vspace{-1pt}
            \item \ok{Inverse Fourier Transform}: $\mathcal{F}^{-1}\{\widehat{f}\}\approx f(x_n)= \sum_{k=0}^{N-1} \widehat{f}_k e^{2\pi i k n / N}$
        \end{enumerate}

        \vspace{1pt}
        \quad \imp{Property}: $\widehat{f}'_k = (ik) \widehat{f}_k$ \quad $\Leftrightarrow$ \quad $\mathcal{F}\{f'\}=(ik)\mathcal{F}\{f\}$
        
        \vspace{1pt}
        \quad \ps{Remark}: Quickly compute DFT: Fast Fourier Transform (FFT) 
    \end{MyTh}

    \begin{MyTh}{\imp{Using Fourier to solve PDE}} \tiny
        For PDE with \ps{periodic BCs}
        
        \vspace{-2pt}
        \begin{enumerate}
            \item Take Fourier Transform in PDE (in space) \qquad \scalebox{0.9}{e.g. $u_t=u_{xx}$ $\to$ $\frac{d}{dt}\widehat{u}_k = -k^2 \widehat{u}_k$}
            \vspace{-2pt}
            \item Take Fourier Transform in IC (in space) \qquad \scalebox{0.9}{e.g. $u(x,0)=f(x)$ $\to$ $\widehat{u}_k(t=0)=$ $\widehat{f}_k$}
            \vspace{-2pt}
            \item Using Transformed PDE and IC \ $\to$ \ a. Directly Solve ODE. \quad b. numerical method for ODE.
            \vspace{-2pt}
            \item Once $\widehat{u}_k$ is obtained, take Inverse Fourier Transform to get $u(x,t)$.
        \end{enumerate}
        
    \end{MyTh}
