    \section{Definitions, Classification of PDEs}

    \subsection{Big O, Classification}

    \begin{MyTh}{Big $\Ocal$}
        $f(t)=\Ocal(g(t))$ as $t\to t_0$ if \ \ \ps{$\exists \eps>0,C>0$ s.t. $\forall |t-t_0|<\eps$, $|f(t)|\leq C|g(t)|$}

        \qquad {\tiny $\Leftrightarrow$ \ps{$\lim_{t\to t_0}[f(t)/g(t)]$ bdd}} \qquad \ps{ps: If not say $t\to t_0$, assume $t\to 0$}
    \end{MyTh}

    \begin{MyTh}{More on Big $\Ocal$}\tiny
        \ok{a.} \ps{$F(x)=G(x)+\Ocal(H(x))$} \quad if $F(x)-G(x)=\Ocal(H(x))$

        \quad \ok{b.} \ps{$F(x)=\Ocal(G(x))+\Ocal(H(x))$} \quad if $F(x)=\Ocal(\max\{|G(x)|,|H(x)|\})$

        \quad \ok{c.} \ps{$F(x,y)=\Ocal(x^m) + \Ocal(y^n)$ \quad if \scalebox{0.9}{[Given $G(x)=\Ocal(x^{\frac{m}{n}})$ $\implies$ $F(x,G(x))=\Ocal(x^m)$], 且另一项同理}}
    \end{MyTh}

    \begin{MyTh}{Classification} \tiny
        For \ps{$au_{xx}+bu_{xy}+cu_{yy}+du_x+eu_y+fu+g=0$} \quad \scalebox{0.9}{\tiny $(a,b,c)\ne(0,0,0)$}

        \vspace{-2pt}
        \begin{enumerate}
            \item \ok{Hyperbolic}: $b^2-4ac>0$ \ \ \ps{Admits non-trivial plane \textit{wave solutions} (Propagation)}
            \vspace{-4pt}
            \item \ok{Parabolic}: $b^2-4ac=0$ \ \ \ps{and we can get \ $2cd=be$ \ or \ $2ae=bd$}
            \vspace{-4pt}
            \item \ok{Elliptic}: $b^2-4ac<0$ \ \ \ps{\textit{No} non-trivial plane \textit{wave solutions} (No Propagation)}
        \end{enumerate}
        
        \quad \ok{Remark}: \ps{\tiny 对变量$x,y$进行缩放后不会改变PDE类型}
    \end{MyTh}

    \subsection{Norm}

    \begin{MyTh}{Weighted 2-Norm}
        $\|\mathbf{u}\|_{2,\Delta x}=\sqrt{(\Delta x)^d}\|u\|_2$ \quad where $d$ s.t. $\mathbf{u}:\R^d\to\R^n$

        \quad - For $d=1$ and $\Delta x=\frac{D}{N}$, then: \ps{$\forall i,u_i=\Ocal((\Delta x)^p)$ \ $\Rightarrow$ \ $\|\mathbf{u}\|_{2,\Delta x}=\Ocal((\Delta x)^p)$}

        \quad - For $A$ symmetric: \imp{$\|A^{-1}\mathbf{w}\|_{2}\leq \frac{1}{|\lambda_{1}|}\|\mathbf{w}\|_{2}$} \quad \scalebox{0.8}{\tiny \ps{$0<|\lambda_{1}|\leq\cdots\leq|\lambda_{n}|$}}

        \quad \imp{Property}: \tiny If $\|\mathbf{u}\|_{2}\leq \|\mathbf{w}\|_{2}$, then $\|\mathbf{u}\|_{2,\Delta x}\leq \|\mathbf{w}\|_{2,\Delta x}$
    \end{MyTh}

    \begin{MyTh}{New Norm}\tiny 
        \ps{The norm $\|\mathbf{u}\|:=\max_{n}\|\mathbf{u}^n\|_{2,\Delta x}$ \quad where $\mathbf{u}^n=[u_1^n,u_2^n,...,u_{M-1}^n]^\top$}

        \qquad\qquad\qquad\quad Which means: Infinity norm in time, weighted 2-norm in space. \quad \scalebox{0.9}{\ps{主要用于双变量PDE}}

        \quad - For $\tau^n_m=\Ocal((\Delta x)^a) + \Ocal((\Delta t)^b)$, $\Delta x=\frac{b-a}{M}$, $\Delta t=\frac{T}{N}$, assume $\Delta t=K(\Delta x)^\beta$, then:
        
        \quad \ \ $\max_{n}\|\underline{\tau}^n\|_{2,\Delta x}=\Ocal((\Delta x)^{\min\{a,b\beta\}})$
    \end{MyTh}

    \subsection{Linear Algebra}

    \begin{MyTh}{Definite}\tiny
        \ok{Positive Definite}: $\lambda_i>0,\forall i$ \quad \ok{Negative Definite}: $\lambda_i<0,\forall i$
    \end{MyTh}
    
    \begin{MyTh}{Symmetric}\tiny
        If $A$ \ok{symmetric}, then \ps{$A=Q\Lambda Q^{\top}$} where $Q$ orthogonal, $\Lambda$ diagonal with eigenvalues.
    \end{MyTh}

    \begin{MyTh}{Normal}\tiny
        Real matrix $A$ is \ok{normal} if \ps{$AA^{\top}=A^{\top}A$} \quad The following are properties of it:
        \begin{enumerate}
            \item $A$ is \ps{orthogonally diagonalizable}, i.e. \ps{$A=Q\Lambda Q^{\top}$} where $Q$ orthogonal, $\Lambda$ diagonal with eigenvalues.
            \vspace{-2pt}
            \item Exists an eigenvectors of $A$: $\{\mathbf{v}_i\}$ s.t. \ps{$\mathbf{v}_i^{*}\mathbf{v}_j=\delta_{ij}$} \qquad\qquad and \ \ \ps{Symmetric $\Rightarrow$ Normal}
        \end{enumerate}
    \end{MyTh}

    \section{Global/Local Error, Convergence, Consistency, Stability}

    \begin{MyTh}{\imp{Exact \& Approx. Sol}}\tiny
        \ok{$U_n$: approx. $n$-th time-step sol}; \quad \ok{$u_n$: exact sol at $n$-th time-step}.
    \end{MyTh}

    \begin{MyTh}{Global Error $(e_n,\mathbf{e}(\Delta t))$}\tiny
        \ok{$e_n=U_n-u(t=n\Delta t)$} \quad \ok{$\mathbf{e}(\Delta t)=[e_0,e_1,\ldots,e_N]^{\top}$}
    \end{MyTh}

    \begin{MyTh}{Convergence}\tiny
        A numerical method is convergent if \ok{$\lim_{\Delta t\to 0}\|\mathbf{e}(\Delta t)\|=0$}
        
        \qquad If \ok{$\lim_{\Delta t\to 0}\frac{\|\mathbf{e}(\Delta t)\|}{(\Delta t)^p}=C$} \ \ (i.e. $\|\mathbf{e}(\Delta t)\|=\Ocal((\Delta t)^p)$) \ $\Rightarrow$ \ \ok{$p$-th order accurate}
    \end{MyTh}

    \begin{MyTh}{Local Truncation Error $(\tau_n)$}\tiny
        数值分析的\textit{离散}迭代公式, \textit{将$U_n$替换为$u_n$后的误差(LHS - RHS)}.
        
        \qquad - e.g. \ For \scalebox{0.9}{\ps{$\frac{U_{n+1}-U_n}{\Delta t}=f(U_n)$}, \ \ \ps{$\tau_{n+1}=\frac{u_{n+1}-u_n}{\Delta t}-f(u_n)$}} \quad $\tau_0=0$

        \qquad - e.g. 我们可以把线性数值方法写成$Au=b$, $A$ matrix, $u=[U_1,...,U_{N-1}]^\top$.

        \qquad\qquad \ \ Then: \ps{$\tau=A u_{\text{exact}}-b$} where \ok{$u_{\text{exact}}=[u_1,...,u_{N-1}]^\top$}, \ok{$\tau=[\tau_1,...,\tau_{N-1}]^\top$}
        
        \qquad \imp{ps: $Ae=-\tau$ where $e=u-u_{\text{exact}}$} \qquad Define \imp{$\tau(\Delta t)=[\tau_1,...,\tau_{N-1}]^\top$}
    \end{MyTh}

    \begin{MyTh}{Consistency}\tiny
        If \ok{$\lim_{\Delta t\to 0}\|\tau(\Delta t)\|=0$}, the method is consistent.
    \end{MyTh}

    \begin{MyTh}{Linear Equation}\tiny
        Goal: Finally change ODE/PDE into \ok{$Au=b$} where \ok{$A$ is matrix}, \ok{$u=[U_1,...,U_{N-1}]^\top$}
    \end{MyTh}

    \begin{MyTh}{Stability} \tiny
        A numerical method is stable if \quad $\exists \Delta x_0$ s.t. For the Linear System \ok{$Au=b$}:
        \begin{enumerate}
            \item $A(\Delta x)$ is invertible $\forall \Delta x<\Delta x_0$
            \item \ok{$\|[A(\Delta x)]^{-1}\mathbf{w}\|\leq C\|\mathbf{w}\|$} \quad for all $\mathbf{w}\in\R^{N-1}$
        \end{enumerate}
        \quad \ok{Property}: Consistent + Stable $\Rightarrow$ Convergent \scalebox{0.9}{\tiny (same norm!)} \quad and $\|\mathbf{e}(\Delta t)\|=\Ocal(\|\tau(\Delta t)\|)$
    \end{MyTh}
    }}

    \section{Intro \& Method of Lines}

    \begin{MyTh}{\imp{Taylor Series}}\tiny
        \scalebox{0.9}{$f(t+\Delta t)=f(t)+f'(t)\Delta t + \frac{1}{2}f''(t)(\Delta t)^2 + \cdots + \Ocal((\Delta t)^N)$} \ \ \tiny \imp{ps:$u_{n+1}=u(n\Delta t+\Delta t)$}
        
        \qquad\qquad\qquad\qquad \scalebox{0.9}{$f(t)=f(a)+f'(a)(t-a)+\frac{f''(a)}{2!}(t-a)^2+\cdots+\frac{f^{(N-1)}(a)}{(N-1)!}(t-a)^N+\Ocal((t-a)^{N})$}
    \end{MyTh}

    \begin{MyTh}{Euler Methods}\tiny
        Forward: $\frac{du}{dt}|_{n\Delta t}\approx \frac{U_{n+1}-U_n}{\Delta t}$; \quad Backward: $\frac{du}{dt}|_{n\Delta t}\approx \frac{U_{n}-U_{n-1}}{\Delta t}$
    \end{MyTh}

    \begin{MyTh}{Leapfrog}\tiny
        $\frac{du}{dt}|_{n\Delta t}\approx \frac{U_{n+1}-U_{n-1}}{2\Delta t}$
    \end{MyTh}
    \ \
    \begin{MyTh}{Trapezoidal}\tiny
        $\frac12[\frac{du}{dt}|_{n\Delta t}+\frac{du}{dt}|_{(n+1)\Delta t}]\approx \frac{U_{n+1}-U_n}{\Delta t}$
    \end{MyTh}

    \begin{MyTh}{Centred Difference}\tiny
        $\frac{d^2u}{dt^2}|_{n\Delta t}\approx \frac{U_{n-1}-2U_n+U_{n+1}}{(\Delta t)^2}$
    \end{MyTh}

    \begin{MyTh}{\imp{Method of Lines}}\tiny
        Using $U_m(t)\approx u(a+m\Delta x,t)$, $U_m^n\approx u(a+m\Delta x,n\Delta t)$ \quad \scalebox{0.8}{\ps{\tiny $a$为空间起点,不重要}}

        \vspace{-2pt}
        \begin{enumerate}
            \item \ok{For 1D:} \ok{a.} 先对空间$x$离散化,使用$U_m$表达, \qquad \scalebox{0.9}{此时 $\frac{d\mathbf{U}}{dt}=...$ where $\mathbf{U}=[U_1(t),U_2(t),...,U_{M-1}(t)]^\top$}
            
            \vspace{-4pt}
            \qquad\quad \ \ \ok{b.} 对时间$t$使用ODE离散化,使用$U_m^n$表达, 得到 Linear Equation.
            
            \vspace{-2pt}
            \item Example: Crank-Nicolson for Heat Eq. $u_t=\kappa u_{xx}$: 对$x$用centred difference; 对$t$用trapezoidal:
            \scalebox{0.9}{\tiny $
                \frac{U_m^{n+1}-U_m^n}{\Delta t} = \frac{\kappa}{2}\left(\frac{U_{m+1}^{n+1}-2U_m^{n+1}+U_{m-1}^{n+1}}{(\Delta x)^2} + \frac{U_{m+1}^n - 2U_m^n + U_{m-1}^n}{(\Delta x)^2}\right)
            $} \\
            \scalebox{0.9}{\tiny $
                \Rightarrow -r U_{m-1}^{n+1} + (2+2r) U_m^{n+1} - r U_{m+1}^{n+1} = r U_{m-1}^n + (2-2r) U_m^n + r U_{m+1}^n, \quad r=\frac{\kappa \Delta t}{(\Delta x)^2}
            $} \\
        \end{enumerate}

        \vspace{-4pt}
        \quad \ok{Solve it}: By iteration: e.g. $A\mathbf{U}^{n+1}=\mathbf{B}\mathbf{U}^n$, \quad where $\mathbf{U}^n=[U_1^n,U_2^n,...,U_{M-1}^n]^\top$.
    \end{MyTh}

    \begin{MyTh}{Truncation for MoL}\tiny 对于 ${\tau}_m$, 离散$x$后$U_m$替换为$u_m$; 对于 $\tau_m^n$, 再离散$t$后$U_m^n$替换为$u_m^n$.

        \vspace{1pt}
        \quad e.g. For Crank-Nicolson above: $\tau_m = \frac{\partial u_m}{\partial t} - \kappa \frac{u_{m+1}-2u_m+u_{m-1}}{(\Delta x)^2}$

        \vspace{-2pt}
        \quad e.g. For Crank-Nicolson above: $\tau_m^n = \frac{u_m^{n+1}-u_m^n}{\Delta t} - \frac{\kappa}{2}(\cdots)$
    \end{MyTh}

    }}

    \section{ Lax-Richtmyer Stability, von Neumann Analysis}

    \begin{MyTh}{Abs. Stability for ODEs}\tiny
        For ODEs \ok{$\frac{d\mathbf{u}}{dt}=A \mathbf{u}$}, $\mathbf{u}$ vector. \quad \scalebox{0.9}{A 1-step method for $t$ \ (i.e. $\mathbf{U}^{n+1}=B\mathbf{U}^n$)}

        \quad It's \ps{absolutely stable} if \quad $\|B\mathbf{w}\|_2 \leq \|\mathbf{w}\|_2$ \quad for all $\mathbf{w}\in\R^{N-1}$
        
        \quad \ok{Property}: If $B$ normal, then absolutely stable $\Leftrightarrow$ \ps{$|\lambda_i(B)|\leq1,\forall i$}

        \quad \ok{Property}: Absolute stability $\Rightarrow$ Lax-Richtmyer stability in \ps{2-norm} and \ps{weighted 2-norm}
    \end{MyTh}

    \begin{MyTh}{\imp{Lax-Richtmyer Stability}}\tiny
        For PDE which discrete form $\frac{1}{\Delta t}\mathbf{U}^{n+1} = \frac{1}{\Delta t}B\mathbf{U}^n + \mathbf{F}^n$, $B$ fixed, $\mathbf{F}$ can change.

        \quad It's \ps{Lax-Richtmyer stable} if \quad $\|B^n\mathbf{w}\| \leq C\|\mathbf{w}\|$ \quad for all $\mathbf{w}\in\R^{N-1}$, $n\leq \frac{T}{\Delta t}=N$

        \quad \ok{Property}: If consistent with $\max_{n}\|\underline{\tau}^n\|_{2,\Delta x}=\Ocal((\Delta x)^\beta)$ and L-R stable $\|B^n\mathbf{w}\|_{2,\Delta x} \leq C\|\mathbf{w}\|_{2,\Delta x}$,

        \quad then, it convergents with $\max_{n}\|\mathbf{e}^n\|_{2,\Delta x}=\Ocal((\Delta x)^\beta)$ \quad \ps{Remark}: \scalebox{0.85}{\ok{Always Assume $\Delta t = K(\Delta x)^\alpha$}}
    \end{MyTh}

    \begin{MyTh}{\imp{von Neumann Analysis}}\tiny
        PDE: \ok{1.} \textit{periodic} BCs in spatial $x$, length $D$. \ok{2.} 满足 L-R 中PDE框架, \ok{3.} $\mathbf{F}^n=0$

        \quad $\Rightarrow$ Let \ps{$v_{k,m}=\exp(2\pi i k \frac{m\Delta x}{D})$} for $m=0,1,...,M-1$. \quad Then vector $\mathbf{v}_k$ 为 $B$ 的\ps{正交特征向量}.

        \quad $\star$ \ \ By \ps{$U^n_m = \exp(2\pi i k \frac{m\Delta x}{D})$, \ $U^{n+1}_m=\alpha_k U^n_m$}, 可得特征值\scalebox{0.9}{(放大因子)} $\alpha_k$ 表达式 (i.e. $B\mathbf{v}_k=\alpha_k \mathbf{v}_k$).

        \quad $\Rightarrow$ If all $\alpha_k$ for $k=0,1,...,M-1$ can be found $\Rightarrow$ $B$ is normal and then:

        \qquad \ If $|\alpha_k|\leq 1,\forall k$ \ $\Rightarrow$ \ps{$B$ is absolutely stable} and \ps{L-R stable in weighted 2-norm}
        
        \quad \ok{Remark}: \ps{假设$\Delta t=K(\Delta x)^\alpha$会得$K$, $\alpha$的限制条件, \quad || \quad For $\mathbf{F}^n\ne0$, 讨论$\mathbf{F}^n=0$足够.}
    \end{MyTh}

    }}

    \section{Boundary Condition}

    \begin{MyTh}{2nd ODE} \tiny
        For $u_xx=\beta(x)$ with $x\in(a,b)$, there are three types of BCs:
        \begin{enumerate}
            \item \ok{Two Dirichlet}: $u(x=a)=c$ \quad $u(x=b)=d$
            \vspace{-2pt}
            \item \ok{One Dirichlet, One Neumann}: $u(x=a)=c$ \quad $u_x(x=b)=d$ \quad or \ \ \scalebox{0.8}{$u_x(x=a)=c$ \ \ $u(x=b)=d$}
            \vspace{-2pt}
            \item \ok{One Neumann, One Constraint}: $u_x(x=a)=c$ \quad $\int_a^b u(x)dx=d$
        \end{enumerate}
    \end{MyTh}

    \vspace{-2pt}
    \begin{MyTh}{Heat PDE} \tiny
        For $u_t=\kappa u_{xx}$ with $x\in(a,b)$, there are three types of BCs: \quad \ok{$\star$ Basic IC}: $u(x,t_0)=P(x)$
        \begin{enumerate}
            \item \ok{Two Dirichlet}: $u(x=a,t)=c$ \quad $u(x=b,t)=d$
            \vspace{-2pt}
            \item \ok{One Dirichlet, One Neumann}: \scalebox{0.95}{$u(x=a,t)=c$ \ \ \ $u_x(x=b,t)=d$} \quad or \ \ \scalebox{0.8}{$u_x(x=a,t)=c$ \ \ $u(x=b,t)=d$}
            \vspace{-2pt}
            \item \ok{Two Neumann}: $u_x(x=a,t)=c$ \quad $u_x(x=b,t)=d$
        \end{enumerate}
        
    \end{MyTh}

    \begin{MyTh}{\ok{Remark}} \tiny
        For $u_x(x=a, t)=c$ $\Rightarrow$ descrite $\tfrac{u_1^n-u_0^n}{\Delta x}=c$ \quad \scalebox{0.9}{\ps{Boundary condition itself can effect the accuracy}}
        
        \quad For $u_t(x, t_0)=P(x)$ $\Rightarrow$ descrite $\tfrac{u_m^1-u_m^0}{\Delta t}=P(a+m\Delta x)$
    \end{MyTh}

    \section{One Time Solve Two Variables PDEs}

    \begin{MyTh}{Elliptic Equation} \tiny
        $Lu = F(x_1,...,x_d)$ is an elliptic PDE if \ \ $L$ is an elliptic operator.

        \quad \ok{Theorem}: $L=\sum_{r=1}^d\sum{s=1}^d A_{r,s}\frac{\partial^2}{\partial x_r \partial x_s} + \sum_{n=1}^d B_n \frac{\partial}{\partial x_n} + C$ is elliptic operator if:

        \vspace{2pt}
        \quad \qquad \qquad \quad \ps{A is symmetric} \ \ and \ \ \ps{A is positive|negative definite ($\lambda_i>0|\lambda_i<0$).}
    \end{MyTh}

    \begin{MyTh}{Poisson Equation} \tiny
        $\Delta u = F(x_1,...,x_d)$ is a Poisson equation \ and \ it's elliptic. \ \ with BCs:

        \quad \ps{Dirichlet}: $u=u_D$ on $\partial\Omega$; \quad \ps{Neumann}: $\Delta u \cdot \hat{\mathbf{n}}=g$ on $\partial\Omega$ where $\int_{\Omega} F = \int_{\partial\Omega} g$ and $\int_{\Omega} u = \int_{\partial\Omega} u_D$. 

        \quad e.g. For 2D: $u_{xx}+u_{yy}=F(x,y)$ with BCs: $u(x,y)=u_D$ on $\partial\Omega$

        \qquad $\to$ \ok{5-point stencil}: $\tfrac{U_{m-1,p}+U_{m+1,p}+U_{m,p-1}+U_{m,p+1}-4U_{m,p}}{(\Delta x)^2}=F(a+m\Delta x, a+p\textcolor{green}{\Delta x})$

        \qquad $\to$ Using $\mathbf{u} = [U_{1,1}, U_{2,1}, ..., U_{M-1,1}, U_{1,2}, ..., U_{M-1,M-1}]^\top$ and 
        
        \qquad \quad \ \ $\mathbf{b}_{m+(p-1)(M-1)}=F(a+m\Delta x, a+p\textcolor{green}{\Delta x})$ then $A\mathbf{u}=\mathbf{b}$ where $A$ is a symmetric matrix.

        \qquad $\to$ \scalebox{0.9}{Eigenvalue/vectors of $A$: $\mathbf{v}_{k,l}=[v_{k,l,1,1}, v_{k,l,2,1}, ..., v_{k,l,1,2}, ..., v_{k,l,M-1,M-1}]^\top$}

        \qquad \quad \ \ \scalebox{0.87}{$v_{k,l,m,p}=\sin(\pi k m \Delta x)\sin(\pi l p \Delta x)$, \quad $\lambda_{k,l}=\frac{2}{(\Delta x)^2}[\cos(\pi k \Delta x)+\cos(\pi l \Delta x)-2]<0$, \ \ \ $\lambda_{\min}=\lambda_{1,1}$}
    \end{MyTh}

    \vspace{1pt}
    \begin{MyTh}{\imp{Remark}} \tiny
        这一节都是用同一个$\Delta x$ 离散$x$和$y$的, 和前面$t,x$分别不同.
    \end{MyTh}


    \begin{MyTh}{Consistency} \tiny
        If $\lim_{\Delta x\to 0}\|\underline{\tau}(\Delta x)\|=0$, the method is consistent. 
    \end{MyTh}

    \begin{MyTh}{Convergence} \tiny
        If $\|\mathbf{e}(\Delta x)\|=\Ocal((\Delta x)^p)$, the method is $p$-th order accurate.
    \end{MyTh}

    \begin{MyTh}{Stability} \tiny
        A numerical method is stable if for the linear system $A\mathbf{u}=\mathbf{b}$:

        \vspace{-2pt}
        \begin{enumerate}
            \item $A(\Delta x)$ is invertible $\forall \Delta x<\Delta x_0$
            \vspace{-2pt}
            \item \ok{$\|[A(\Delta x)]^{-1}\mathbf{w}\|\leq C\|\mathbf{w}\|$} \quad for all $\mathbf{w}\in\R^{N-1}$
        \end{enumerate}

        \ok{Property}: If $A$ normal, then $\|A^{-1}\mathbf{w}\|_2 \leq \frac{1}{|\lambda_1|}\|\mathbf{w}\|_2$ where $0<|\lambda_1|\leq\cdots\leq|\lambda_{N-1}|$ are eigenvalues of $A$.
    \end{MyTh}

    \section{Spectral Methods}
    
    \begin{MyTh}{Inner Product Space for Function} \tiny
        $\langle \phi_{i}, \phi_j \rangle = \int_a^b \phi_i \ \phi_j \  w \  dx$ \quad ps: $w(x)$ is weight function.

        \quad For Fourier Series: $w(x)=1$, $a=-\pi$, $b=\pi$: \quad $\sin(nx)$, $\cos(mx)$ are orthogonal basis.
    \end{MyTh}

    \begin{MyTh}{Fourier Series} \tiny
        $f(x)=\frac{a_0}{2}+\sum_{n=1}^{\infty} a_n \cos(nx) + b_n \sin(nx)=\sum_{n=-\infty}^{\infty} A_{k} e^{ikx}$ \quad where
        
        \vspace{2pt}
        \quad $\to$ $a_0=\frac{1}{\pi}\int_{-\pi}^{\pi} f(x) dx$ \qquad $a_n=\frac{1}{\pi}\int_{-\pi}^{\pi} f(x)\cos(nx) dx$ \qquad $b_n=\frac{1}{\pi}\int_{-\pi}^{\pi} f(x)\sin(nx) dx$.

        \vspace{2pt}
        \quad $\to$ \scalebox{0.9}{$k>0$: $A_k=\frac{1}{2}(a_k - i b_k)$; \quad $k<0$: $A_k=\frac{1}{2}(a_{-k} + i b_{-k})$; \quad $k=0$: $A_0=\frac{a_0}{2}$. \quad For real func: \ps{$A_{-k}=\overline{A_k}$}}

        \vspace{2pt}
        \quad \imp{Property}: $A_k=\frac{1}{2\pi}\int_{-\pi}^{\pi} f(x) e^{-ikx}dx$ \quad and \quad $f'(x)=\sum_{k=-\infty}^{\infty} (ik) A_k e^{ikx}$
    \end{MyTh}
    
    \begin{MyTh}{More on Fourier Series} \tiny
        Assume $f$ is periodic with period $2\pi$, then:

        \begin{enumerate}
            \item The Fourier Transform: $\mathcal{F}\{f(x)\}=\widehat{f}(k)=\frac{1}{2\pi}\int_{-\pi}^{\pi} f(x) e^{-ikx} dx$
            \item The inverse Fourier Transform: $\mathcal{F}^{-1}\{\widehat{f}(k)\}=f(x)=\sum_{k=-\infty}^{\infty} \widehat{f}(k) e^{ikx}$
        \end{enumerate}

        \quad \imp{Change $L$ $\to$ $2\pi$ period}: If $f(x)$ has period $L$ \ \ $\Rightarrow$ \ \ $g(x)=f(\frac{L}{2\pi}y)$ has $2\pi$ period.
    \end{MyTh}

    \begin{MyTh}{\imp{Discrete Fourier Transform (DFT)}} \tiny
        \scalebox{0.9}{If $x_n=n\Delta x$ \ok{equally-spaced} with $n=0,1,...,N-1$; $\Delta x=\frac{2\pi}{N}$}:

        \vspace{-1pt}
        \begin{enumerate}
            \item \ok{Fourier Transform}: $\mathcal{F}\{f\}=\widehat{f}_k = \frac{2\pi}{N}\sum_{n=0}^{N-1} f(x_n) e^{-i k x_n}$ \quad \scalebox{0.9}{$k=-\frac{N}{2}+1,...,\frac{N}{2}$}
            \item \ok{Inverse FT}: $\mathcal{F}^{-1}\{\widehat{f}\} = f(x_n) = \frac{1}{2\pi}$ \scalebox{0.8}{$\sum_{k=-N/2+1}^{N/2}$} $ \widehat{f}_k e^{i k x_n} = \frac{1}{2\pi}$ \scalebox{0.8}{$\sum_{k=-N/2}^{N/2}$} ${}^{\prime}\!\widehat{f}_k e^{i k x_n}$
        \end{enumerate}

        \quad where \scalebox{0.9}{${}^{\prime}\!\widehat{f}_k = \widehat{f}_k$ for $k=-\frac{N}{2}+1,...,\frac{N}{2}-1$; ${}^{\prime}\!\widehat{f}_{-N/2}=\widehat{g}_{N/2}=\frac{1}{2}\widehat{f}_{N/2}$}.
        
        \quad \imp{Property}: $\widehat{f'}_k = (ik)\widehat{f}_k \quad \Leftrightarrow \quad \mathcal{F}\{f'\}= \left(ik\right)\mathcal{F}\{f\}$

        \vspace{1pt}
        \quad \ps{Remark}: Quickly compute DFT: Fast Fourier Transform (FFT)
    \end{MyTh}

    \begin{MyTh}{\imp{Using Fourier to solve PDE}} \tiny
        For PDE with \ps{periodic BCs}
        
        \vspace{-2pt}
        \begin{enumerate}
            \item Take Fourier Transform in PDE (in space) \qquad \scalebox{0.9}{e.g. $u_t=u_{xx}$ $\to$ $\frac{d}{dt}\widehat{u}_k = -k^2 \widehat{u}_k$}
            \vspace{-2pt}
            \item Take Fourier Transform in IC (in space) \qquad \scalebox{0.9}{e.g. $u(x,0)=f(x)$ $\to$ $\widehat{u}_k(t=0)=$ $\widehat{f}_k$}
            \vspace{-2pt}
            \item Using Transformed PDE and IC \ $\to$ \ a. Directly Solve ODE. \quad b. numerical method for ODE. $\to$ $\widehat{u}_k(t)$
            \vspace{-2pt}
            \item Inverse Fourier Transform to get $u(x_n,t)$ 对$x_n$点$t$的估计或精确解.
        \end{enumerate}        
    \end{MyTh}

            \item DFT: \scalebox{0.9}{$\widehat{f}_k = \frac{2\pi}{N}\sum_{n=0}^{N-1} f(x_n) e^{-i k x_n}$} \ \ $\Rightarrow$ \ \ $p(x)=\frac{1}{2\pi}$\scalebox{0.8}{$\sum_{k=-N/2}^{N/2}$}${}^{\prime}\!\widehat{f}_k e^{i k x}$
            
            \vspace{-2pt}

            \vspace{-2pt}

            \vspace{-3pt}
            $\Rightarrow$ \scalebox{0.9}{$P_{\delta}(x-x_m)=\frac{1}{2\pi}$\scalebox{0.8}{$\sum_{k=-N/2}^{N/2}$}${}^{\prime}\!\widetilde{\delta}_{k-m} e^{i k x}=\frac{1}{2\pi}$\scalebox{0.8}{$\sum_{k=-N/2}^{N/2}$}${}^{\prime}\!e^{i k (x-x_m)}$} and \scalebox{0.95}{$P_{\delta}(x)=\frac{\sin(\pi x/\Delta x)}{(2\pi/\Delta x) \tan(x/2)}$}

            \vspace{-2pt}

            \vspace{-2pt}

            \vspace{-4pt}

    \section{The Steepest Descent Method}

    \begin{MyTh}{Theorem} \tiny
        Let $\phi(\mathbf{x})=\frac{1}{2}\mathbf{x}^TA\mathbf{x}-\mathbf{b}^T\mathbf{x}$, then it has unique minimum if $A$ is \textit{positive definite}.

        \quad If $A$ is symmetric positive definite \quad $\Rightarrow$ \quad \ps{$\mathbf{x} \text{:} A\mathbf{x}=\mathbf{b} \Leftrightarrow \mathbf{x} \text{:} \phi(\mathbf{x})=\min_{\widetilde{\mathbf{x}}}\phi(\widetilde{\mathbf{x}})=\min_{\widetilde{\mathbf{x}}}[\frac{1}{2}\widetilde{\mathbf{x}}^TA\widetilde{\mathbf{x}}-\mathbf{b}^T\widetilde{\mathbf{x}}]$}
    \end{MyTh}

    \imp{Following need $A$ to be symmetric positive definite.}

    \begin{MyTh}{Steepest Descent Method} \tiny
        To solve \ps{$\min \phi(x)$} or \ps{$A\mathbf{x}=\mathbf{b}$}, we can use the following iteration:

        \quad $\mathbf{x}^{(k)}=\mathbf{x}^{(k-1)}+\alpha_{k-1} \nabla \phi(\mathbf{x}^{(k-1)})$ \quad and \quad $\alpha_{k-1}$ s.t. $\min_{\alpha} \phi(\mathbf{x}^{(k-1)}-\alpha \nabla \phi(\mathbf{x}^{(k-1)}))$.

        \begin{enumerate}
            \item \ok{Property 1}: $\nabla \phi(\mathbf{x})$ is the direction of steepest ascent (most rapid increase) of $\phi$ at $\mathbf{x}$.
            \item \ok{Property 2}: $\alpha_{k-1} = \frac{(\mathbf{r}^{(k-1)})^{\top}\mathbf{r}^{(k-1)}}{(\mathbf{r}^{(k-1)})^{\top}A\mathbf{r}^{(k-1)}}$ \quad and \quad $\nabla \phi(\mathbf{x}^{(k)}) = A\mathbf{x}^{(k)}-\mathbf{b} =: -\mathbf{r}^{(k)}$ is \ok{$k$-th residual}.
            \vspace{-1pt}
            \item \ok{Property 3}: $\mathbf{r}^{(k)} = \mathbf{r}^{(k-1)} - \alpha_{k-1} A \mathbf{r}^{(k-1)}$
        \end{enumerate}

        \quad \ps{Remark}: Convergence behaviour is \textit{highly problem-dependent}. \scalebox{0.9}{(如果$\phi$的等高线是圆形则快,椭圆形则慢)}
    \end{MyTh}

    \begin{MyTh}{The Conjugate Gradient Method} \tiny
        To solve \ps{$\min \phi(x)$} or \ps{$A\mathbf{x}=\mathbf{b}$}, we can use the following iteration:

        \quad $\mathbf{x}^{(k)}=\mathbf{x}^{(k-1)}+\alpha_{k-1} \mathbf{p}^{(k-1)}$ \quad and \quad $\alpha_{k-1}$ s.t. $\min_{\alpha} \phi(\mathbf{x}^{(k-1)}-\alpha \mathbf{p}^{(k-1)})$ 
        
        \quad and $\mathbf{p}^{(k)}$ is \textit{$A$-conjugate} to previous \ok{search directions}: $\mathbf{p}^{(k-1)}$: \ok{$(\mathbf{p}^{(k)})^{\top}A\mathbf{p}^{(k-1)}=0$}

        \vspace{-2pt}
        \begin{enumerate}
            \item \ok{Property 1}: $\alpha_{k-1} = \frac{(\mathbf{r}^{(k-1)})^{\top}\mathbf{r}^{(k-1)}}{(\mathbf{p}^{(k-1)})^{\top}A\mathbf{p}^{(k-1)}}$ \quad and \quad $-\mathbf{r}^{(k)} = A\mathbf{x}^{(k)}-\mathbf{b}$ is \ok{$k$-th residual}.
            \vspace{-3pt}
            \item \ok{Property 2}: $\mathbf{p}^{(k)} = \mathbf{r}^{(k)} + \beta_{k-1} \mathbf{p}^{(k-1)}$ \quad where $\beta_{k-1} = \frac{(\mathbf{r}^{(k)})^{\top}\mathbf{r}^{(k)}}{(\mathbf{r}^{(k-1)})^{\top}\mathbf{r}^{(k-1)}}$
            \vspace{-2pt}
            \item \ok{Property 3}: $(\mathbf{p}^{(k)})^{\top}A\mathbf{p}^{(j)}=0$ \quad and \quad $(\mathbf{r}^{(k)})^{\top}\mathbf{r}^{(j)}=0$ \quad for $j=0,1,...,k-1$
            \item \ok{Property 4}: \scalebox{0.9}{$\mathrm{span}\{\mathbf{p}^{(0)}, ..., \mathbf{p}^{(k-1)}\} = \mathrm{span}\{A^0\mathbf{r}^{(0)}, ..., A^{k-1}\mathbf{r}^{(0)}\}=\mathrm{span}\{A\mathbf{e}^{(0)}, ..., A^k\mathbf{e}^{(0)}\}$}

            \vspace{-3pt}
            \scalebox{0.9}{where $\mathbf{e}^{(k)}=\mathbf{x}^{(k)}-\mathbf{x}$ is the error at $k$-th iteration.} \qquad \qquad \qquad \scalebox{0.9}{(Quicker than JAC|GAU)}
            \vspace{-3pt}
            \item \ok{Exact Property}: For $A\in\mathbb{R}$ has $n$ \textit{distinct} eigenvalues $\Rightarrow$ solve in \ok{at most $n$} iterations. (By exact arithmetic)
        \end{enumerate}
    \end{MyTh}

    \begin{MyTh}{A-Norm}
        The $A$-norm of $\mathbf{x}$ is defined as $\|\mathbf{x}\|_A = \sqrt{\mathbf{x}^TA\mathbf{x}}$.
    \end{MyTh}

    \begin{MyTh}{Condition Number} \tiny
        The condition number of $A$ is $\kappa(A) = \|A\|\|A^{-1}\|\geq 1$ \quad and \quad $\kappa_2(A) = \frac{\lambda_{\max}(A)}{\lambda_{\min}(A)}$
    \end{MyTh}

    \begin{MyTh}{Error of Conjugate Gradient} \tiny

    \end{MyTh}
