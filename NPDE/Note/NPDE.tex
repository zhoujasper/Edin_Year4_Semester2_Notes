    \section{Definitions, Classification of PDEs}

    \subsection{Big O, Classification}

    \begin{MyTh}{Big $\Ocal$}
        $f(t)=\Ocal(g(t))$ as $t\to t_0$ if \ \ \ps{$\exists \eps>0,C>0$ s.t. $\forall |t-t_0|<\eps$, $|f(t)|\leq C|g(t)|$}

        \qquad {\tiny $\Leftrightarrow$ \ps{$\lim_{t\to t_0}[f(t)/g(t)]$ bdd}} \qquad \ps{ps: If not say $t\to t_0$, assume $t\to 0$}
    \end{MyTh}

    \begin{MyTh}{More on Big $\Ocal$}\tiny
        \ok{a.} \ps{$F(x)=G(x)+\Ocal(H(x))$} \quad if $F(x)-G(x)=\Ocal(H(x))$

        \quad \ok{b.} \ps{$F(x)=\Ocal(G(x))+\Ocal(H(x))$} \quad if $F(x)=\Ocal(\max\{|G(x)|,|H(x)|\})$

        \quad \ok{c.} \ps{$F(x,y)=\Ocal(x^m) + \Ocal(y^n)$ \quad if \scalebox{0.9}{[Given $G(x)=\Ocal(x^{\frac{m}{n}})$ $\implies$ $F(x,G(x))=\Ocal(x^m)$], 且另一项同理}}
    \end{MyTh}

    \begin{MyTh}{Classification} \tiny
        For \ps{$au_{xx}+bu_{xy}+cu_{yy}+du_x+eu_y+fu+g=0$} \quad \scalebox{0.9}{\tiny $(a,b,c)\ne(0,0,0)$}

        \vspace{-2pt}
        \begin{enumerate}
            \item \ok{Hyperbolic}: $b^2-4ac>0$ \ \ \ps{Admits non-trivial plane wave solutions}
            \vspace{-4pt}
            \item \ok{Parabolic}: $b^2-4ac=0$ \ \ \ps{and we can get \ $2cd=be$ \ or \ $2ae=bd$}
            \vspace{-4pt}
            \item \ok{Elliptic}: $b^2-4ac<0$ \ \ \ps{No non-trivial plane wave solutions}
        \end{enumerate}
        
        \quad \ok{Remark}: \ps{\tiny 对变量$x,y$进行缩放后不会改变PDE类型}
    \end{MyTh}

    \subsection{Norm}

    \begin{MyTh}{Weighted 2-Norm}
        $\|\mathbf{u}\|_{2,\Delta x}=\sqrt{(\Delta x)^d}\|u\|_2$ \quad where $d$ s.t. $\mathbf{u}:\R^d\to\R^n$

        \quad - For $d=1$ and $\Delta x=\frac{D}{N}$, then: \ps{$\forall i,u_i=\Ocal((\Delta x)^p)$ \ $\Rightarrow$ \ $\|\mathbf{u}\|_{2,\Delta x}=\Ocal((\Delta x)^p)$}

        \quad - For $A$ symmetric: \ps{$\|A^{-1}\mathbf{w}\|_{2,\Delta x}\leq \frac{1}{|\lambda_{1}|}\|\mathbf{w}\|_{2,\Delta x}$} \quad \scalebox{0.8}{\tiny \ps{$0<|\lambda_{1}|\leq\cdots\leq|\lambda_{n}|$}}

        \quad \ok{Property}: \tiny If $\|\mathbf{u}\|_{2}\leq \|\mathbf{w}\|_{2}$, then $\|\mathbf{u}\|_{2,\Delta x}\leq \|\mathbf{w}\|_{2,\Delta x}$
    \end{MyTh}

    \begin{MyTh}{New Norm}\tiny 
        \ps{The norm $\|\mathbf{u}\|:=\max_{n}\|\mathbf{u}^n\|_{2,\Delta x}$ \quad where $\mathbf{u}^n=[u_1^n,u_2^n,...,u_{M-1}^n]^\top$}

        \qquad\qquad\qquad\quad Which means: Infinity norm in time, weighted 2-norm in space. \quad \scalebox{0.9}{\ps{主要用于双变量PDE}}

        \quad - For $\tau^n_m=\Ocal((\Delta x)^a) + \Ocal((\Delta t)^b)$, $\Delta x=\frac{b-a}{M}$, $\Delta t=\frac{T}{N}$, assume $\Delta t=K(\Delta x)^\beta$, then:
        
        \quad \ \ $\max_{n}\|\underline{\tau}^n\|_{2,\Delta x}=\Ocal((\Delta x)^{\min\{a,b\beta\}})$
    \end{MyTh}

    \subsection{Linear Algebra}

    \begin{MyTh}{Definite}\tiny
        \ok{Positive Definite}: $\lambda_i>0,\forall i$ \quad \ok{Negative Definite}: $\lambda_i<0,\forall i$
    \end{MyTh}
    
    \begin{MyTh}{Symmetric}\tiny
        If $A$ \ok{symmetric}, then \ps{$A=Q\Lambda Q^{\top}$} where $Q$ orthogonal, $\Lambda$ diagonal with eigenvalues.
    \end{MyTh}

    \begin{MyTh}{Normal}\tiny
        Real matrix $A$ is \ok{normal} if \ps{$AA^{\top}=A^{\top}A$} \quad The following are properties of it:
        \begin{enumerate}
            \item $A$ is \ps{orthogonally diagonalizable}, i.e. \ps{$A=Q\Lambda Q^{\top}$} where $Q$ orthogonal, $\Lambda$ diagonal with eigenvalues.
            \vspace{-2pt}
            \item Exists an eigenvectors of $A$: $\{\mathbf{v}_i\}$ s.t. \ps{$\mathbf{v}_i^{*}\mathbf{v}_j=\delta_{ij}$} \qquad\qquad and \ \ \ps{Symmetric $\Rightarrow$ Normal}
        \end{enumerate}
    \end{MyTh}
    \vspace{-2pt}

    \section{Global/Local Error, Convergence, Consistency, Stability}

    \begin{MyTh}{\imp{Exact \& Approx. Sol}}\tiny
        \ok{$U_n$: approx. $n$-th time-step sol}; \quad \ok{$u_n$: exact sol at $n$-th time-step}.
    \end{MyTh}

    \begin{MyTh}{Global Error $(e_n,\mathbf{e}(\Delta t))$}\tiny
        \ok{$e_n=U_n-u(t=n\Delta t)$} \quad \ok{$\mathbf{e}(\Delta t)=[e_0,e_1,\ldots,e_N]^{\top}$}
    \end{MyTh}

    \begin{MyTh}{Convergence}\tiny
        A numerical method is convergent if \ok{$\lim_{\Delta t\to 0}\|\mathbf{e}(\Delta t)\|=0$}
        
        \qquad If \ok{$\lim_{\Delta t\to 0}\frac{\|\mathbf{e}(\Delta t)\|}{(\Delta t)^p}=C$} \ \ (i.e. $\lim_{\Delta t\to 0}\|\mathbf{e}(\Delta t)\|=\Ocal((\Delta t)^p)$) \ $\Rightarrow$ \ok{$p$-th order accurate}
    \end{MyTh}

    \begin{MyTh}{Local Truncation Error $(\tau_n)$}\tiny
        数值分析的\textit{离散}迭代公式, \textit{将$U_n$替换为$u_n$后的误差(LHS - RHS)}.
        
        \qquad - e.g. \ For \scalebox{0.9}{\ps{$\frac{U_{n+1}-U_n}{\Delta t}=f(U_n)$}, \ \ \ps{$\tau_{n+1}=\frac{u_{n+1}-u_n}{\Delta t}-f(u_n)$}} \quad $\tau_0=0$

        \qquad - e.g. 我们可以把线性数值方法写成$Au=b$, $A$ matrix, $u=[U_1,...,U_{N-1}]^\top$.

        \qquad\qquad \ \ Then: \ps{$\tau=A u_{\text{exact}}-b$} where \ok{$u_{\text{exact}}=[u_1,...,u_{N-1}]^\top$}, \ok{$\tau=[\tau_1,...,\tau_{N-1}]^\top$}
        
        \qquad \imp{ps: $Ae=-\tau$ where $e=u-u_{\text{exact}}$} \qquad Define \imp{$\tau(\Delta t)=[\tau_1,...,\tau_{N-1}]^\top$}
    \end{MyTh}

    \begin{MyTh}{Consistency}\tiny
        If \ok{$\lim_{\Delta t\to 0}\|\tau(\Delta t)\|=0$}, the method is consistent.
    \end{MyTh}

    \begin{MyTh}{Linear Equation}\tiny
        Goal: Finally change ODE/PDE into \ok{$Au=b$} where \ok{$A$ is matrix}, \ok{$u=[U_1,...,U_{N-1}]^\top$}
    \end{MyTh}

    \begin{MyTh}{Stability}
        A numerical method is stable if \quad $\exists \Delta x_0$ s.t. For the Linear System \ok{$Au=b$}:
        \begin{enumerate}
            \item $A(\Delta x)$ is invertible $\forall \Delta x<\Delta x_0$
            \item \ok{$\|A^{-1}(\Delta x)\mathbf{w}\|\leq C\|\mathbf{w}\|$} \quad for all $\mathbf{w}\in\R^{N-1}$
        \end{enumerate}
        \quad \ok{Property}: Consistent + Stable $\Rightarrow$ Convergent \scalebox{0.9}{\tiny (same norm!)} \quad and $\|\mathbf{e}(\Delta t)\|=\Ocal(\|\tau(\Delta t)\|)$
    \end{MyTh}
    }}

    \section{Intro \& Method of Lines}

    \begin{MyTh}{\imp{Taylor Series}}\tiny
        \scalebox{0.9}{$f(t+\Delta t)=f(t)+f'(t)\Delta t + \frac{1}{2}f''(t)(\Delta t)^2 + \cdots + \Ocal((\Delta t)^N)$} \ \ \tiny \imp{ps:$u_{n+1}=u(n\Delta t+\Delta t)$}
        
        \qquad\qquad\qquad\qquad \scalebox{0.9}{$f(t)=f(a)+f'(a)(t-a)+\frac{f''(a)}{2!}(t-a)^2+\cdots+\frac{f^{(N-1)}(a)}{(N-1)!}(t-a)^N+\Ocal((t-a)^{N})$}
    \end{MyTh}

    \begin{MyTh}{Euler Methods}\tiny
        Forward: $\frac{du}{dt}|_{n\Delta t}\approx \frac{U_{n+1}-U_n}{\Delta t}$; \quad Backward: $\frac{du}{dt}|_{n\Delta t}\approx \frac{U_{n}-U_{n-1}}{\Delta t}$
    \end{MyTh}

    \begin{MyTh}{Leapfrog}\tiny
        $\frac{du}{dt}|_{n\Delta t}\approx \frac{U_{n+1}-U_{n-1}}{2\Delta t}$
    \end{MyTh}
    \ \
    \begin{MyTh}{Trapezoidal}\tiny
        $\frac12[\frac{du}{dt}|_{n\Delta t}+\frac{du}{dt}|_{(n+1)\Delta t}]\approx \frac{U_{n+1}-U_n}{\Delta t}$
    \end{MyTh}

    \begin{MyTh}{Centred Difference}\tiny
        $\frac{d^2u}{dt^2}|_{n\Delta t}\approx \frac{U_{n-1}-2U_n+U_{n+1}}{(\Delta t)^2}$
    \end{MyTh}

    \begin{MyTh}{\imp{Method of Lines}}\tiny
        Using $U_m(t)\approx u(a+m\Delta x,t)$, $U_m^n\approx u(a+m\Delta x,n\Delta t)$ \quad \scalebox{0.8}{\ps{\tiny $a$为空间起点,不重要}}

        \vspace{-2pt}
        \begin{enumerate}
            \item \ok{For 1D:} \ok{a.} 先对空间$x$离散化,使用$U_m$表达, \qquad \scalebox{0.9}{此时 $\frac{d\mathbf{U}}{dt}=...$ where $\mathbf{U}=[U_1(t),U_2(t),...,U_{M-1}(t)]^\top$}
            
            \vspace{-4pt}
            \qquad\quad \ \ \ok{b.} 对时间$t$使用ODE离散化,使用$U_m^n$表达, 得到 Linear Equation.
            
            \vspace{-2pt}
            \item Example: Crank-Nicolson for Heat Eq. $u_t=\kappa u_{xx}$: 对$x$用centred difference; 对$t$用trapezoidal:
            \scalebox{0.9}{\tiny $
                \frac{U_m^{n+1}-U_m^n}{\Delta t} = \frac{\kappa}{2}\left(\frac{U_{m+1}^{n+1}-2U_m^{n+1}+U_{m-1}^{n+1}}{(\Delta x)^2} + \frac{U_{m+1}^n - 2U_m^n + U_{m-1}^n}{(\Delta x)^2}\right)
            $} \\
            \scalebox{0.9}{\tiny $
                \Rightarrow -r U_{m-1}^{n+1} + (2+2r) U_m^{n+1} - r U_{m+1}^{n+1} = r U_{m-1}^n + (2-2r) U_m^n + r U_{m+1}^n, \quad r=\frac{\kappa \Delta t}{(\Delta x)^2}
            $} \\
        \end{enumerate}

        \vspace{-4pt}
        \quad \ok{Solve it}: By iteration: e.g. $A\mathbf{U}^{n+1}=\mathbf{B}\mathbf{U}^n$, \quad where $\mathbf{U}^n=[U_1^n,U_2^n,...,U_{M-1}^n]^\top$.
    \end{MyTh}

    \begin{MyTh}{Truncation for MoL}\tiny 对于 ${\tau}_m$, 离散$x$后$U_m$替换为$u_m$; 对于 $\tau_m^n$, 再离散$t$后$U_m^n$替换为$u_m^n$.

        \vspace{1pt}
        \quad e.g. For Crank-Nicolson above: $\tau_m = \frac{\partial u_m}{\partial t} - \kappa \frac{u_{m+1}-2u_m+u_{m-1}}{(\Delta x)^2}$

        \vspace{-2pt}
        \quad e.g. For Crank-Nicolson above: $\tau_m^n = \frac{u_m^{n+1}-u_m^n}{\Delta t} - \frac{\kappa}{2}(\cdots)$
    \end{MyTh}

    }}

    \section{ Lax-Richtmyer Stability, von Neumann Analysis}

    \begin{MyTh}{Abs. Stability for ODEs}\tiny
        For ODEs \ok{$\frac{d\mathbf{u}}{dt}=A \mathbf{u}$}, $\mathbf{u}$ vector. \quad \scalebox{0.9}{A 1-step method for $t$ \ (i.e. $\mathbf{U}^{n+1}=B\mathbf{U}^n$)}

        \quad It's \ps{absolutely stable} if \quad $\|B\mathbf{w}\|_2 \leq \|\mathbf{w}\|_2$ \quad for all $\mathbf{w}\in\R^{N-1}$
        
        \quad \ok{Property}: If $B$ normal, then absolutely stable $\Leftrightarrow$ \ps{$|\lambda_i(B)|\leq1,\forall i$}

        \quad \ok{Property}: Absolute stability $\Rightarrow$ Lax-Richtmyer stability in \ps{2-norm} and \ps{weighted 2-norm}
    \end{MyTh}

    \begin{MyTh}{\imp{Lax-Richtmyer Stability}}\tiny
        For PDE which discrete form $\frac{1}{\Delta t}\mathbf{U}^{n+1} = \frac{1}{\Delta t}B\mathbf{U}^n + \mathbf{F}^n$, $B$ fixed, $\mathbf{F}$ can change.

        \quad It's \ps{Lax-Richtmyer stable} if \quad $\|B^n\mathbf{w}\| \leq C\|\mathbf{w}\|$ \quad for all $\mathbf{w}\in\R^{N-1}$, $n\leq \frac{T}{\Delta t}=N$

        \quad \ok{Property}: If consistent with $\max_{n}\|\underline{\tau}^n\|_{2,\Delta x}=\Ocal((\Delta x)^\beta)$ and L-R stable $\|B^n\mathbf{w}\|_{2,\Delta x} \leq C\|\mathbf{w}\|_{2,\Delta x}$,

        \quad then, it convergents with $\max_{n}\|\mathbf{e}^n\|_{2,\Delta x}=\Ocal((\Delta x)^\beta)$ \quad \ps{Remark}: \scalebox{0.85}{\ok{Always Assume $\Delta t = K(\Delta x)^\alpha$}}
    \end{MyTh}

    \begin{MyTh}{\imp{von Neumann Analysis}}\tiny
        PDE: \ok{1.} \textit{periodic} BCs in spatial $x$, length $D$. \ok{2.} 满足 L-R 中PDE框架, \ok{3.} $\mathbf{F}^n=0$

        \quad $\Rightarrow$ Let \ps{$v_{k,m}=\exp(2\pi i k \frac{m\Delta x}{D})$} for $m=0,1,...,M-1$. \quad Then vector $\mathbf{v}_k$ 为 $B$ 的\ps{正交特征向量}.

        \quad $\star$ \ \ By \ps{$U^n_m = \exp(2\pi i k \frac{m\Delta x}{D})$, \ $U^{n+1}_m=\alpha_k U^n_m$}, 可得特征值\scalebox{0.9}{(放大因子)} $\alpha_k$ 表达式 (i.e. $B\mathbf{v}_k=\alpha_k \mathbf{v}_k$).

        \quad $\Rightarrow$ If all $\alpha_k$ for $k=0,1,...,M-1$ can be found $\Rightarrow$ $B$ is normal and then:

        \qquad \ If $|\alpha_k|\leq 1,\forall k$ \ $\Rightarrow$ \ps{$B$ is absolutely stable} and \ps{L-R stable in weighted 2-norm}
        
        \quad \ok{Remark}: \ps{假设$\Delta t=K(\Delta x)^\alpha$会得$K$, $\alpha$的限制条件, \quad || \quad For $\mathbf{F}^n\ne0$, 讨论$\mathbf{F}^n=0$足够.}
    \end{MyTh}

    }}

    \section{Boundary Condition}

    \begin{MyTh}{2nd ODE} \tiny
        For $u_xx=\beta(x)$ with $x\in(a,b)$, there are three types of BCs:
        \begin{enumerate}
            \item \ok{Two Dirichlet}: $u(x=a)=c$ \quad $u(x=b)=d$
            \vspace{-2pt}
            \item \ok{One Dirichlet, One Neumann}: $u(x=a)=c$ \quad $u_x(x=b)=d$ \quad or \ \ \scalebox{0.8}{$u_x(x=a)=c$ \ \ $u(x=b)=d$}
            \vspace{-2pt}
            \item \ok{One Neumann, One Constraint}: $u_x(x=a)=c$ \quad $\int_a^b u(x)dx=d$
        \end{enumerate}
    \end{MyTh}

    \vspace{-2pt}
    \begin{MyTh}{Heat PDE} \tiny
        For $u_t=\kappa u_{xx}$ with $x\in(a,b)$, there are three types of BCs: \quad \ok{$\star$ Basic IC}: $u(x,t_0)=P(x)$
        \begin{enumerate}
            \item \ok{Two Dirichlet}: $u(x=a,t)=c$ \quad $u(x=b,t)=d$
            \vspace{-2pt}
            \item \ok{One Dirichlet, One Neumann}: \scalebox{0.95}{$u(x=a,t)=c$ \ \ \ $u_x(x=b,t)=d$} \quad or \ \ \scalebox{0.8}{$u_x(x=a,t)=c$ \ \ $u(x=b,t)=d$}
            \vspace{-2pt}
            \item \ok{Two Neumann}: $u_x(x=a,t)=c$ \quad $u_x(x=b,t)=d$
        \end{enumerate}
        
    \end{MyTh}

    \begin{MyTh}{\ok{Remark}} \tiny
        For $u_x(x=a, t)=c$ $\Rightarrow$ descrite $\tfrac{u_1^n-u_0^n}{\Delta x}=c$ \quad \scalebox{0.9}{\ps{Boundary condition itself can effect the accuracy}}
        
        \quad For $u_t(x, t_0)=P(x)$ $\Rightarrow$ descrite $\tfrac{u_m^1-u_m^0}{\Delta t}=P(a+m\Delta x)$
    \end{MyTh}

    \section{Appendix \qquad \imp{\tiny $x,x^2,x^3$ as $x\to0$, \quad $x^{-1},x^{-2},x^{-3}$ as $x\to\text{inf}$}}
    \tiny
    
    \begin{MyTh}{Types of PDEs}\tiny
        Laplace Eq.: $u_{xx}+u_{yy}=0$ {\tiny (E)} \quad Poisson Eq.: $u_{xx}+u_{yy}=f(x,y)$ {\tiny (E)}

        \quad Helmholtz Eq.: $u_{xx}+u_{yy}+\alpha u=0,\alpha\ne0$ {\tiny (E)} \quad Heat Eq.: $u_t=\kappa(u_{xx}+u_{yy}),\kappa>0$ {\tiny (P)}

        \quad Wave Eq.: $u_{xx}-u_{yy}=f(x,y)$ {\tiny (H)} \quad Advection Eq.: $u_t+vy_x+wu_y=0$ {\tiny (H)}
    \end{MyTh}

    \begin{MyTh}{Taylor Series} \tiny
        \scalebox{0.9}{$f(t+\Delta t)=f(t)+f'(t)\Delta t + \frac{1}{2}f''(t)(\Delta t)^2 + \cdots + \Ocal((\Delta t)^N)$} \ \ \tiny \imp{ps:$u_{n+1}=u(n\Delta t+\Delta t)$}
        
        \quad \scalebox{0.9}{$u^{n+1}_m = u^n_m + \Delta t (u_t)^n_m + \frac{(\Delta t)^2}{2}(u_{tt})^n_m + \cdots $} \quad where \ps{$(u_t)^n_m = \frac{\partial u}{\partial t}(a+m\Delta x,n\Delta t)$}

        \quad \scalebox{0.9}{$u_{m+1}^n = u_m^n + \Delta x (u_x)_m^n + \frac{(\Delta x)^2}{2}(u_{xx})_m^n + \cdots $} \quad where \ps{$(u_x)_m^n = \frac{\partial u}{\partial x}(a+m\Delta x,n\Delta t)$}
    \end{MyTh}

    \vspace{1pt}
    \begin{MyTh}{Useful Tools}\tiny
        If \ps{$e^n=Ae^{n-1}+B\tau^n$}, then \ \ \ps{$e^n=A^ne^0+\sum_{k=1}^{n}A^{n-k}B\tau^k$}, \quad $A$: amplification factor.
    \end{MyTh}

    \begin{MyTh}{Others}
        Periodic boundary conditions for $x$: \ps{$u(a,t)=u(b,t)$, $u_x(a,t)=u_x(b,t)$} \quad $a,b$ 边界.
    \end{MyTh}
