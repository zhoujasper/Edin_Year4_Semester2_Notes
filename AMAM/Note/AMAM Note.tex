    \section{Notation \& Perturbation Method}

    \begin{MyTh}{Big $\Ocal$}
        $f(z)=\Ocal(g(z))$ {\tiny as $z\to z_0$} \ if \ok{$\lim\limits_{z\to z_0}\frac{f(z)}{g(z)}$ bdd} \ $\Leftrightarrow$ \ \ok{$|f(z)|\leq A|g(z)|$ {\tiny as $z\to z_0$}}
    \end{MyTh}

    \vspace{-4pt}
    \begin{MyTh}{Small $o$}
        $f(z)=o(g(z))$ {\tiny as $z\to z_0$} \ if \ok{$\lim\limits_{z\to z_0}\frac{f(z)}{g(z)}=0$} \ $\Leftrightarrow$ \ \ok{$f(z)\ll g(z)$ {\tiny as $z\to z_0$}}
    \end{MyTh}

    \vspace{-4pt}
    \begin{MyTh}{Asymptotic $\sim$}
        $f(z)\sim g(z)$ {\tiny as $z\to z_0$} \ if \ok{$\lim\limits_{z\to z_0}\frac{f(z)}{g(z)}=1$}
    \end{MyTh}

    \vspace{-8pt}
    \hspace{245pt} \scalebox{0.8}{\tiny (as $z\to z_0$)}

    \begin{MyTh}{Asymptotic Sequence}
        $\{\varphi_n(z)\}$: If $\forall n\in\N$ \ok{$\varphi_{n+1}(z)=o(\varphi_n(z))$} {\tiny (or \ps{$\varphi_{n+1}(z)\ll \varphi_n(z)$}, $\downarrow$)}
    \end{MyTh}

    \vspace{-1pt}
    \begin{MyTh}{Asymptotic Expansion}
        {\tiny \scalebox{0.9}{$f(z)\sim \sum\limits_{n=0}^{M}a_n\varphi_n(z)$}} {\tiny as $z\to z_0$} \ if \ok{\tiny \scalebox{0.9}{$f(z)-\sum\limits_{n=0}^{N}a_n\varphi_n(z)\ll\varphi_N(z)$,$\forall N\leq M$}}
