    \section{Notation \& Perturbation Method}

    \begin{MyTh}{Big $\Ocal$}\tiny
        $f(z)=\Ocal(g(z))$ \ as $z\to z_0$ \quad if \ok{$\lim\limits_{z\to z_0}\frac{f(z)}{g(z)}$ bdd} \ $\Leftrightarrow$ \ \ok{$|f(z)|\leq A|g(z)|$} \ as $z\to z_0$
    \end{MyTh}

    \vspace{-2pt}
    \begin{MyTh}{Small $o$}\tiny
        $f(z)=o(g(z))$ \ as $z\to z_0$ \quad if \ok{$\lim\limits_{z\to z_0}\frac{f(z)}{g(z)}=0$} \ $\Leftrightarrow$ \ \ok{$f(z)\ll g(z)$} \ as $z\to z_0$
    \end{MyTh}

    \vspace{-2pt}
    \begin{MyTh}{Asymptotic $\sim$}\tiny
        $f(z)\sim g(z)$ \ as $z\to z_0$ \ if \ok{$\lim\limits_{z\to z_0}\frac{f(z)}{g(z)}=1$}
    \end{MyTh}

    \vspace{-1pt}
    \begin{MyTh}{Asymptotic Sequence}\tiny
        Seq $\{\varphi_n(z)\}$ is asy. if \ \ $\forall n\in\N$ \ok{$\varphi_{n+1}(z)=o(\varphi_n(z))$} \ as $z\to z_0$
    \end{MyTh}

    \begin{MyTh}{Asymptotic Expansion}\tiny
        \scalebox{0.9}{$f(z)\sim \sum\limits_{n=0}^{M}a_n\varphi_n(z)$} \ as $z\to z_0$ to $M$ \quad if \ok{$\phi_n(z)$ is asy. seq.} \ and
        
        \vspace{-4pt}
        \quad \scalebox{0.9}{\ok{$f(z)-\sum\limits_{n=0}^{N}a_n\varphi_n(z)=o(\varphi_N(z))=\Ocal(\varphi_{N+1}(z))$}}, \ $\forall N\leq M$ \ as $z\to z_0$

        \vspace{-3pt}
        \quad \scalebox{0.9}{\ok{$f(x,\eps)\sim \sum\limits_{n=0}^{\infty}a_n(x)\eps^n$} \ps{ uniformly} \ in $x\in I$ \ as $\eps\to 0$ \ \ if \ok{$\eps$与$x$无关}\ps{(i.e. $\sup\limits_{x\in I}|f(x,\eps)-\sum\limits_{n=0}^{N}a_n(x)\eps^n|=o(\eps^N)$)}}
    \end{MyTh}

    \begin{MyTh}{\imp{Singular|Regular - Perturbation Method}}
        {\tiny \ok{a.} Let $\eps=0$ solve $f(x,0)=0$.}

        \quad {\tiny \ok{b.i.} 如果$f$结构变了\scalebox{0.9}{(阶数降低)}, Let $x=\eps^{-\alpha}X$, PDB 找$\alpha$ \ \ \ok{c.} 假设$X=a_0+a_1\eps^{\beta}+a_2\eps^{2\beta}+\cdots$ 代入求解}

        \quad {\tiny \ok{b.ii.} 如果$f$结构不变/变, 但有重根$x_0$, Let $x=x_0+\eps^{\alpha}X$, PDB 找$\alpha$.\qquad\qquad $\uparrow$ $\beta$看化简后的$X$方程得出}

        \quad {\tiny \ok{Principle of dominant balance (PDB)}: For Singular Perturbation $p(x,\eps)=0$, Let $x=\eps^{\alpha}X$, 代入选择$\alpha$:}

        \quad {\tiny 对于$\eps$来说 \ \ \ok{a.} 至少有两项阶数相同且最低;  \ \ \ \ok{b.} 其他项阶数均更高. (即可忽略)}
    \end{MyTh}
    }}

    \begin{MyTh}{Taylor}\tiny
        \scalebox{0.9}{$f(x)=\sum\limits_{n=0}^{N-1}\frac{f^{(n)}(x_0)}{n!}(x-x_0)^n + \frac{f^{(N)}(\xi)}{N!}(x-x_0)^N$} \ $\Rightarrow$ \ \scalebox{0.9}{\ok{$f(x)\sim \sum\limits_{n=0}^{\infty}\frac{f^{(n)}(x_0)}{n!}(x-x_0)^n$} as $x\to x_0$}
    \end{MyTh}

    \vspace{-3pt}
    \begin{MyTh}{Exp. Integral}
        \tiny $E_1(x)=\int_{x}^{\infty}\frac{e^{-t}}{t}dt$ , \ \ \ps{\tiny ($x\to 0$发散)} \quad \ok{$E_1(x)\sim e^{-x}\sum\limits_{n=0}^{\infty}(-1)^n\frac{n!}{x^{n+1}}$, $x\to\infty$}
    \end{MyTh}

    \vspace{-3pt}
    \begin{MyTh}{Gamma Func}
        \tiny $\Gamma(x)=\int_{0}^{\infty}t^{x-1}e^{-t}dt$ \quad \ok{$\Gamma(x+1)=x\Gamma(x)$} and \ok{$\Gamma(n)=(n-1)!$} \quad \ps{$\Gamma(\frac{1}{2})=\sqrt{\pi}$}
    \end{MyTh}

    \begin{MyTh}{Incomplete Gamma Func}
        {\tiny \ok{Lower}: $\gamma(a,x)=\int_0^{x}t^{a-1}e^{-t}dt$ \quad \ok{Upper}: $\Gamma(a,x)=\int_x^{\infty}t^{a-1}e^{-t}dt$}

        \vspace{-2pt}
        \ \ - \ok{$\gamma(a,x)=\Gamma(a)-\Gamma(a,x)$} \qquad {\tiny \ok{$\Gamma(a,x) \sim e^{-x}x^{a-1}\sum\limits_{n=0}^{\infty}\frac{\Gamma(a)}{\Gamma(a-n)}x^{-n}$}, $x\to\infty$}
    \end{MyTh}

        \ok{b.} $f'(t_0)=0$, $f''(t_0)>0$  \quad \ok{c.} Taylor of $f(t)=f(t_0)+\frac{f''(t_0)}{2}(t-t_0)^2+\cdots$ \\
        
        \vspace{-7pt}\quad

    \end{MyTh}
