    \section{Notation \& Perturbation Method}

    \begin{MyTh}{Big $\Ocal$}\tiny
        $f(z)=\Ocal(g(z))$ \ as $z\to z_0$ \quad if \ok{$\lim\limits_{z\to z_0}\frac{f(z)}{g(z)}$ bdd} \ $\Leftrightarrow$ \ \ok{$|f(z)|\leq A|g(z)|$} \ as $z\to z_0$
    \end{MyTh}

    \vspace{-2pt}
    \begin{MyTh}{Small $o$}\tiny
        $f(z)=o(g(z))$ \ as $z\to z_0$ \quad if \ok{$\lim\limits_{z\to z_0}\frac{f(z)}{g(z)}=0$} \ $\Leftrightarrow$ \ \ok{$f(z)\ll g(z)$} \ as $z\to z_0$
    \end{MyTh}

    \vspace{-2pt}
    \begin{MyTh}{Asymptotic $\sim$}\tiny
        $f(z)\sim g(z)$ \ as $z\to z_0$ \ if \ok{$\lim\limits_{z\to z_0}\frac{f(z)}{g(z)}=1$} \qquad \imp{\tiny ps:$e^{-x}=o(x^{-n})$, $\forall n>0$ \ as $x\to\infty$}
    \end{MyTh}

    \vspace{-1pt}
    \begin{MyTh}{Asymptotic Sequence}\tiny
        Seq $\{\varphi_n(z)\}$ is asy. if \ \ $\forall n\in\N$ \ok{$\varphi_{n+1}(z)=o(\varphi_n(z))$} \ as $z\to z_0$
    \end{MyTh}

    \begin{MyTh}{Asymptotic Expansion}\tiny
        \scalebox{0.9}{$f(z)\sim \sum\limits_{n=0}^{M}a_n\varphi_n(z)$} \ as $z\to z_0$ to $M$ \quad if \ok{$\phi_n(z)$ is asy. seq.} \ and
        
        \vspace{-4pt}
        \quad \scalebox{0.9}{\ok{$f(z)-\sum\limits_{n=0}^{N}a_n\varphi_n(z)=o(\varphi_N(z))=\Ocal(\varphi_{N+1}(z))$}}, \ $\forall N\leq M$ \ as $z\to z_0$

        \vspace{-3pt}
        \quad \scalebox{0.9}{\ok{$f(x,\eps)\sim \sum\limits_{n=0}^{\infty}a_n(x)\eps^n$} \ps{ uniformly} \ in $x\in I$ \ as $\eps\to 0$ \ \ if \ok{$\eps$与$x$无关}\ps{(i.e. $\sup\limits_{x\in I}|f(x,\eps)-\sum\limits_{n=0}^{N}a_n(x)\eps^n|=o(\eps^N)$)}}
    \end{MyTh}

    \begin{MyTh}{\imp{Singular|Regular - Perturbation Method}}
        {\tiny \ok{a.} Let $\eps=0$ solve $f(x,0)=0$.}

        \quad {\tiny \ok{b.i.} 如果$f$结构变了\scalebox{0.9}{(阶数降低)}, Let $x=\eps^{-\alpha}X$, PDB 找$\alpha$ \ \ \ok{c.} 假设$X=a_0+a_1\eps^{\beta}+a_2\eps^{2\beta}+\cdots$ 代入求解}

        \quad {\tiny \ok{b.ii.} 如果$f$结构不变/变, 但有重根$x_0$, Let $x=x_0+\eps^{\alpha}X$, PDB 找$\alpha$.\qquad\qquad $\uparrow$ $\beta$看化简后的$X$方程得出}

        \quad {\tiny \ok{Principle of dominant balance (PDB)}: For Singular Perturbation $p(x,\eps)=0$, Let $x=\eps^{\alpha}X$, 代入选择$\alpha$:}

        \quad {\tiny 对于$\eps$来说 \ \ \ok{a.} 至少有两项阶数相同且最低;  \ \ \ \ok{b.} 其他项阶数均更高. (即可忽略)}
    \end{MyTh}

    \section{\footnotesize Integral By Parts, Laplace's Method \& Method of Steepest Descents}

    \begin{MyTh}{Taylor}\tiny
        \scalebox{0.9}{$f(x)=\sum\limits_{n=0}^{N-1}\frac{f^{(n)}(x_0)}{n!}(x-x_0)^n + \frac{f^{(N)}(\xi)}{N!}(x-x_0)^N$} \ $\Rightarrow$ \ \scalebox{0.9}{\ok{$f(x)\sim \sum\limits_{n=0}^{\infty}\frac{f^{(n)}(x_0)}{n!}(x-x_0)^n$} as $x\to x_0$}
    \end{MyTh}

    \vspace{-3pt}
    \begin{MyTh}{Exp. Integral}
        \tiny $E_1(x)=\int_{x}^{\infty}\frac{e^{-t}}{t}dt$ , \ \ \ps{\tiny ($x\to 0$发散)} \quad \ok{$E_1(x)\sim e^{-x}\sum\limits_{n=0}^{\infty}(-1)^n\frac{n!}{x^{n+1}}$, $x\to\infty$}
    \end{MyTh}

    \vspace{-3pt}
    \begin{MyTh}{Gamma Func}
        \tiny $\Gamma(x)=\int_{0}^{\infty}t^{x-1}e^{-t}dt$ \quad \ok{$\Gamma(x+1)=x\Gamma(x)$} and \ok{$\Gamma(n)=(n-1)!$} \quad \ps{$\Gamma(\frac{1}{2})=\sqrt{\pi}$}
    \end{MyTh}

    \begin{MyTh}{Incomplete Gamma Func}
        {\tiny \ok{Lower}: $\gamma(a,x)=\int_0^{x}t^{a-1}e^{-t}dt$ \quad \ok{Upper}: $\Gamma(a,x)=\int_x^{\infty}t^{a-1}e^{-t}dt$}

        \vspace{-2pt}
        \ \ - \ok{$\gamma(a,x)=\Gamma(a)-\Gamma(a,x)$} \qquad {\tiny \ok{$\Gamma(a,x) \sim e^{-x}x^{a-1}\sum\limits_{n=0}^{\infty}\frac{\Gamma(a)}{\Gamma(a-n)}x^{-n}$}, $x\to\infty$}
    \end{MyTh}

    \vspace{-4pt}
    \begin{MyTh}{\imp{Laplace's Method}}\tiny For \ $I(x)=\int_{a}^{b}e^{xf(t)}g(t)dt$ \ as $x\to\infty$

        \vspace{-8pt}
        \begin{enumerate}
            \item If $f(t)$ has \ps{unique max} at $t=t_0\in(a,b)$ $\Rightarrow$ \scalebox{0.9}{$I\sim \int^{\infty}_{\infty} e^{x[f(t_0)+\frac{f''(t_0)}{2}(t-t_0)^2+\cdots]}g(t_0)dt$} \\
            \scalebox{0.9}{(留n项非常数展开为 $n$ term approximation)} \quad \ps{(ps: $\max$ $\leftrightarrow$ $f'(t_0)=0$, $f''(t_0)<0$)} \\
            \imp{Useful Tool}: Now: \scalebox{0.9}{$I\sim e^{xf(t_0)}\int^{\infty}_{\infty} e^{x[\frac{f''(t_0)}{2}(t-t_0)^2+\cdots]}g(t_0)dt$ \ \ $\Rightarrow$ Let $\tau^2=-[\frac12 f''(t_0)(t-t_0)^2+\cdots]$} \\
            \scalebox{0.98}{Assume $t=\tau+b\tau^2+c\tau^3+\cdots$ as $\tau\to0$ \ \ $\Rightarrow$ \ \ 代入换元式子求系数 \ \ $\Rightarrow$ \ \ $dt=(1+2b\tau+3c\tau^2+\cdots)d\tau$}
            \item \ok{Exact Substitution}: 由1的基础,得到$f(t)$在$t_0$处的展开式,换元令$f(t)$就等于\ps{展开式前几项求解}.
            \item \ok{If $g=g(t,x)$}: $e^{xf(t)}g(t,x)=e^{xf(t)+\ln(g(t,x))}$, 看$xf(t)+\ln(g(t,x))$的最大值,类似1处理. \\
            - If $t_0=f(x)$, 换元令$t=f(x)s$.
        \end{enumerate}
    \end{MyTh}

    \begin{MyTh}{\imp{Method of Steepest Descents}}\tiny
        For $I(x)=\int_{C}e^{xf(z)}g(z)dz$ \ as $x\to\infty$. \quad \ok{\scalebox{0.9}{\tiny $f$ 容易避开singular point}}
        
        \vspace{-3pt}
        \quad If $z_0$ is saddle point: $f'(z_0)=0, f''(z_0)\neq0$ \quad $f(z)$ Taylor at $z_0$: $f(z)=f(z_0)+\frac{f''(z_0)}{2}(z-z_0)^2+\cdots$

        \vspace{-2pt}
        \begin{enumerate}
            \item Let $z=z_0+e^{i\theta}y$ \scalebox{0.8}{($\theta$选让$f''(z_0)e^{2i\theta}$为负实数)} \ $\to$ \ \scalebox{0.9}{$I(x)\sim \int_{-\infty}^{\infty} e^{x[f(z_0)-\frac{|f''(z_0)|}{2}y^2+\cdots]}g(z_0) e^{i\theta}dy$}
            \item If $f$ 不容易避开singular point: \ok{a.}$\mathrm{Im}(f(z))=\mathrm{Im}(f(z_0))$ and $z_0\in C_0$ to find steepest descent path $C_0$
            \item \ok{b.} 画图看哪些singular points 在$C\to C_0$变形路径上, 计算residues. \quad \ok{c.} $I=$ 1 部分 + $2\pi i\sum$ residues.
            \item \ok{More than one saddle point}: \scalebox{0.9}{i. 如果通过积分区域可以舍弃,则舍弃 \ \ ii. saddle points: $z_k$, 取能让$\mathrm{Re}(f(z_k))$最小的.} \\
            \scalebox{0.9}{\ok{iii.} 如果多个saddle points 都满足$\mathrm{Re}(f(z_k))$最小, 则都要计算后叠加.}
            \item \ok{Exact Substitution}: 由1的基础,得到$f(z)$在$z_0$处的展开式,换元令$f(z)$就等于\ps{展开式前几项求解}.
            \item \ok{Remark}: 类似Laplace 3一样,可以对$xf(x)$一个整体处理, 必要时$z_0=f(x)\to$换元$z=f(x)s$.
        \end{enumerate}
    \end{MyTh}

    \begin{MyTh}{Residue}\tiny 
        If $z_0$ pole of order $m$ of $f$, \ok{$\Rightarrow\mathrm{Res}(f,z_0)=\frac{1}{(m-1)!}\lim\limits_{z\to z_0}\frac{d^{m-1}}{dz^{m-1}}[(z-z_0)^mf(z)]$}
    \end{MyTh}

    \vspace{-2pt}
    \begin{MyTh}{Method of Stationary Phase}\tiny
        $\cos(xf(\theta))\to\mathrm{Re}(e^{ixf(\theta)})$ 回到Steepest Descents, 同理$\sin(xf(\theta))$
    \end{MyTh}

    \section{Perturbation Methods in ODEs}

    \begin{MyTh}{Regular Perturbation}
        For \ $L(y,\eps)=0$ , \ let \ $y(x,\eps)=y_0(x)+\eps y_1(x)+\eps^2 y_2(x)+\cdots$

        \quad {\tiny $\Rightarrow$ \ \ok{a.} 代入 $\Ocal(\eps^n)$ \ \ $\to$ \ \ \ok{b.} Boundary/Initial Condition 求解 $y_n(x)$ \ \ $\to$ \ \ \ok{a.} }
    \end{MyTh}

    \begin{MyTh}{\imp{Singular Perturbation}}\tiny
        e.g. \ $\eps y'' + p(x)y' + q(x)y = 0$, \ $y(a)=A$ \  $y(b)=B$, \ $a<b$, \ 大致步骤:
        \begin{enumerate}
            \item Assume boundary layer at $x=a|b$. \ \ Let $x=a+\delta(\eps)X$ or $x=b-\delta(\eps)X$
            \vspace{-2pt}
            \item Assume inner $Y(X)=y(x)$, chain rule 求导代入原ODE \ $\to$ Domain balance 找出 $\delta(\eps)=\eps^{\alpha}$ \\
            \textbf{Recall PDB}: \ok{a.} 至少有两项阶数相同且最低; \quad \ok{b.} 其他项阶数均更高.
            \vspace{-2pt}
            \item Assume $Y=Y_0(X)+\eps^\beta Y_1(X)+\eps^{2\beta} Y_2(X)+\cdots$ \ \ 代入求解 \ \ $\to$ 边界/初始条件 求解 $Y_0(X)$
            \vspace{-2pt}
            \item \ok{Matching}: Consider $X\to\infty|x\to a|b$, \quad If limits not exist (除非X) $\to$ boundary layer 在另一个端点 $\to 1$ \\
            \scalebox{0.85}{If limits exist, boundary layer 正确 $\to$ 渐进outer ($x$), 算inner极限($X$); 算不了保留$X$ $\to$ $x$和$X$的关系代入 $\to$ 比较得系数.}
        \end{enumerate}
        
        \quad \ok{Remark 1}: Outer solution \textit{只能}用在 away from boundary layer 的区域(包括边界点).

        \quad \ok{Remark 2}: Inner solution \textit{只能}用在 boundary layer 附近的区域(包括边界点).

        \quad \ok{Useful Lemma}: For $\eps y'' + p(x)y' + q(x)y = 0$, \ $y(a)=A$ \  $y(b)=B$, \ $a<b$, $p(x)\ne0$ in $[a,b]$ then

        \qquad\qquad\qquad\qquad Has boundary layer at $x=a$ if $p>0$; \ Has boundary layer at $x=b$ if $p<0$.
    \end{MyTh}

    \section{WKB Theory \& Solving PDE}
    \begin{MyTh}{\imp{WKB (I)}}\tiny
        For $u''+k^2q(x)u=0$, \ps{$k\gg 1$} we have: $u\sim A(x)e^{ik\phi(x)}$. \quad Substitution can get: \scalebox{0.9}{\imp{($q$里不能带$k$)}}

        \quad If $q(x)<0$: $u(x)\sim A_{+}|q|^{-\frac{1}{4}}e^{k\int^x_{x_0} \sqrt{|q(t)|}dt} + A_{-}|q|^{-\frac{1}{4}}e^{-k\int^x_{x_0} \sqrt{|q(t)|}dt}$ \quad \scalebox{0.9}{\ok{(Exp. Decay/Growth)}}

        \quad If $q(x)>0$: $u(x)\sim A_{+}q^{-\frac{1}{4}}e^{ik\int^x_{x_0} \sqrt{q(t)}dt} + A_{-}q^{-\frac{1}{4}}e^{-ik\int^x_{x_0} \sqrt{q(t)}dt}$ \quad \scalebox{0.9}{\ok{(Oscillation)}}

        \quad \ok{Turning Point}: If $q(x_0)=0$, then $x_0$ is a WKB turning point. \quad \imp{\scalebox{0.9}{ps: 如果$q(x)$含$k$,对其展开保留到$k^0$项}}
    \end{MyTh}

    \begin{MyTh}{\imp{WKB (II)}}\tiny
        For $u''+k^2q(x)u=0$, \ps{$k\gg 1$} Then:

        \quad If $q(x)<0$: Assume $u\sim A(x)e^{k\phi(x)}$ then: $u'=(A'+k\phi'A)e^{k\phi}$ and
        
        \vspace{1pt}
        \qquad $u''=(A''+2k\phi'A'+k\phi''A+k^2\phi'^2A)e^{k\phi}$ \scalebox{0.9}{代入并比较} $\Ocal(k^2)$, $\Ocal(k)$ \ \ \scalebox{0.9}{(No $\Ocal(1)$)}

        \vspace{2pt}
        \quad If $q(x)>0$: Assume $u\sim A(x)e^{ik\phi(x)}$ then: $u'=(A'+ik\phi'A)e^{ik\phi}$ and
        
        \vspace{1pt}
        \qquad $u''=(A''+2ik\phi'A+k\phi''A-k^2\phi'^2A)e^{ik\phi}$ \scalebox{0.9}{代入并比较} $\Ocal(k^2)$, $\Ocal(k)$ \ \ \scalebox{0.9}{(No $\Ocal(1)$)}
    \end{MyTh}

    \begin{MyTh}{\imp{Method of Characteristics}|IC} \tiny
        For $u_t + c(x,t)u_x = r(x,t,u)$ with IC $u(x,t_0)=f(x)$:
        \begin{enumerate}
            \item In characteristic form: $\tfrac{du}{dt}=r(x,t,u)$ on $\tfrac{dx}{dt}=c(x,t)$ \quad $\to$ \scalebox{0.9}{解ODEs$\to$ $u$,$x$各产生一个未知常数$A$,$B$.}
            \item Assume characteristic curve \scalebox{0.9}{(solution of $\tfrac{dx}{dt}=c(x,t)$)} $x(t=t_0)=\xi$ SO $u(t=t_0)=f(\xi)$: \\
            $\to$ get $u(t=t_0)=f(\xi)$ on $x(t=t_0)=\xi$ \quad $\to$ solve $A,B$.
            \item Substitute $A,B$ back to get $u$.
        \end{enumerate}
    \end{MyTh}

    \begin{MyTh}{Method of Characteristics|BC} \tiny
        For $u_t + c(x,t)u_x = r(x,t,u)$ with BC $u(x_0,t)=f(t)$:
        \begin{enumerate}
            \item In characteristic form: $\tfrac{du}{dt}=r(x,t,u)$ on $\tfrac{dx}{dt}=c(x,t)$ \quad $\to$ \scalebox{0.9}{解ODEs$\to$ $u$,$t$各产生一个未知常数$A$,$B$.}
            \item Assume characteristic curve \scalebox{0.9}{(solution of $\tfrac{dx}{dt}=c(x,t)$)} $t(x=x_0)=\xi$ SO $u(x=x_0)=f(\xi)$: \\
            $\to$ get $u(x=x_0)=f(\xi)$ on $t(x=x_0)=\xi$ \quad $\to$ solve $A,B$.
            \item Substitute $A,B$ back to get $u$.
        \end{enumerate}
    \end{MyTh}

    \begin{MyTh}{Special Cases of Method of Characteristics} \tiny
        \begin{enumerate}
            \item 如果Characteristic curve 和初始曲线/边界条件相交 $\to$ ill-posed \& if not unique $\to$ no solution.
            \item \imp{For $u_t + c(u)u_x = 0$:} If a initial jump $u(x,t_0)=u_L$ for $x<b$ and $u(x,t_0)=u_R$ for $x>b$ \\
            \ps{If $c(u_L)<c(u_R)$} $\to$ expansion wave; \ps{If $c(u_L)>c(u_R)$} $\to$ shock wave.
            \item \imp{For $u_t + c(u)u_x = 0$, $u(x,t_0)=f(x)$:} \scalebox{0.9}{The \ok{breaking time} $t_b=t_0+\min_{\xi}\frac{-1}{F'(\xi)}$ where $F(\xi)=c(f(\xi))$} \\
            \vspace{-2pt}
            Break time 是 $\frac{du}{dx}=\infty$ \scalebox{0.9}{(i.e. 分母为0)} 时的最早时间. \scalebox{0.9}{(通过chain rule得到)} \ \ || \ \ \imp{If $\min_{\xi} F'(\xi)<0 \to $ shock}
        \end{enumerate}
    \end{MyTh}

    \begin{MyTh}{\ok{Conservation Law}} \tiny
        \imp{$u_t + c(u)u_x = 0$}: $u_t + [\int c(u)du]_x = 0$
    \end{MyTh}

    \begin{MyTh}{\imp{Shock Wave}|$u_L,u_R$} \tiny
        For $u_t + c(u)u_x = 0$, $u(x,t_0)=u_L$ for $x<b$ and $u(x,t_0)=u_R$ for $x>b$.
        \begin{enumerate}
            \item 正常解 $\to$ In conservation form: $u_t + [\int c(u)du]_x = 0$ \\
            \ok{shock velocity} $\frac{ds}{dt}=\dot{s}=\frac{[q]}{[\rho]}$ where $q=\int c(u)du$ and $\rho=u$; $[u]:=u_{\text{ahead}}-u_{\text{behind}}$. \\
            其中: $u_{\text{ahead}}=u_R$ , $u_{\text{behind}}=u_L$
            \item Solve $\dot{s}$. As shock starts at $x=b$ at $t=t_0$, $s(t=t_0)=b$ get unknown constant.
            \item Then solution $u=u_L$ for $x<s(t)$ and $u=u_R$ for $x>s(t)$. 
        \end{enumerate}
    \end{MyTh}

    \begin{MyTh}{Shock Wave|$f(x)$} \tiny
        \scalebox{0.92}{For $u_t + c(u)u_x = 0$, $u(x,t_0)=f(x)$ with $t_b=t_0+\min_{\xi}\frac{-1}{F'(\xi)}$ where $F(\xi)=c(f(\xi))$.}
        \begin{enumerate}
            \item 正常解 $\to$ In conservation form: $u_t + [\int c(u)du]_x = 0$ \\
            \ok{shock velocity} $\frac{ds}{dt}=\dot{s}=\frac{[q]}{[\rho]}$ where $q=\int c(u)du$ and $\rho=u$; $[u]:=u_{\text{ahead}}-u_{\text{behind}}$. \\
            其中: $u_{\text{ahead}}=\lim\limits_{x\to s(t)^+}u(x,t)$ , $u_{\text{behind}}=\lim\limits_{x\to s(t)^-}u(x,t)$ \scalebox{0.85}{(which is calculated by characteristics method)}
            \vspace{-2pt}
            \item Solve $\dot{s}$. As shock starts at $x(t=t_b)$ at $t=t_b$, $s(t=t_b)=x(t=t_b)$ get unknown constant. \\
            \ps{ps: i.e. $x(t=t_b)=\xi_{\star}+F(\xi_{\star})(t_b-t_0)$, \ \ 其中$\xi_{\star}$是使得$t_b=t_0+\min_{\xi}\frac{-1}{F'(\xi)}$的$\xi$.}
        \end{enumerate}
        
        \vspace{-2pt}
        \quad \ok{Remark}: \scalebox{0.95}{If $u_t + c(u)u_x = 0$, 大体思路都不变, 除了 In conservation form: $u_t + [\int c(u)du]_x = r(x,t,u)$.}
    \end{MyTh}
    
    \begin{MyTh}{Remark} \tiny
        \scalebox{0.9}{可能Expansion和Shock 冲突 $\to$ 原先Shock \textit{消失}并\textit{留Expansion fan}合并. $\to$ 寻找第一次 $t_b>0$ 可能发生的时间.}

        \quad \scalebox{0.9}{$\to$ 解新 $\dot{s}$; $u_{\text{R}},u_{\text{L}}$ 是\textit{expansion fan值,shock前的值}. $\to$ 用原先$s(t=t_b)$ 解新$s$未知数. $\to$ 新$u$, 留expansion fan.}
    \end{MyTh}

    \section{Sturm-Liouville Theory}
    
    \begin{MyTh}{\imp{Regular S-L}} \tiny
        $\frac{d}{dx}[p(x)\frac{dy}{dx}]+q(x)y+\lambda \rho(x)y=0$, with $a<x<b$ and BCs:
        
        \vspace{2pt} 
        \qquad - $\beta_1 y(a) + \beta_2 y'(a)=0$ and $\beta_3 y(b) + \beta_4 y'(b)=0$.

        \vspace{-2pt}
        \begin{enumerate}
            \item \ok{Eigenvalues}: \ok{a.} All $\lambda$ are \textit{real}; \ \ \ok{b.} Infinite $\lambda$'s: $\lambda_1<\lambda_2<\lambda_3<\cdots$; \ \ \ok{c.} 存在最小不存在最大.
            \vspace{-2pt}
            \item \ok{Eigenfunctions}: \ok{a.} $\lambda_n$ $\leftrightarrow$ $u_n$ \textit{unique}; \ \ \ok{b.} $u_n(x)$ has \textit{$n-1$ zeros in $(a,b)$}; \ \ \ok{c.} $u_n \to$ orthogonal basis.
            \item \ok{Algebra}: \scalebox{0.95}{\ok{a.} $\langle f,g\rangle = \int_a^b f(x)g(x)\rho(x)dx$; \ \ \ok{b.} Any $f(x)$ $\to$ $f(x)=\sum_{n=1}^{\infty}a_nu_n(x)$, \ \ $a_n=\frac{\langle f,u_n\rangle}{\langle u_n,u_n\rangle}$.}
            \vspace{-5pt}
            \item \ok{Rayleigh Quotient}: $\lambda_n = \frac{ [-pu_n u_n']_a^b + \int_a^b (p u_n'^2 - q u_n^2) dx}{\int_a^b \rho u_n^2 dx}$.
            \item \ok{Useful Tools}: $u_n,\lambda_n$ homogeneous eigenvalues/functions. \ \ $\Rightarrow$ \ \ $[pu_n']' + q u_n + \lambda_n \rho u_n=0$.
        \end{enumerate}
    \end{MyTh}

    \begin{MyTh}{How to Solve} \tiny

    \section{Green's Function}
    \ps{\tiny ps: To solve \textit{forced equation}. 即PDE RHS $\ne0$}


    \begin{MyTh}{\imp{Green's Function}} \tiny
        For PDE $L u(x,t) = F(x,t)$ where $L$ is \textit{linear} differential operation with BCs and ICs.
        
        \vspace{-2pt}
        \begin{enumerate}
            \item Consider $G(x,t;\xi,\tau)$ satisfying $L G(x,t;\xi,\tau) = \delta(x-\xi)\delta(t-\tau)$ with BCs and ICs.
            \vspace{-2pt}
            \item Solve $G$ $\to$ $G(x,t;\xi,\tau)$.
            \vspace{-2pt}
            \item Then $u(x,t) = \int_{-\infty}^{\infty} \int_{-\infty}^{\infty} G(x,t;\xi,\tau) F(\xi,\tau) d\xi d\tau$.
        \end{enumerate}

        \vspace{1pt}
        \quad \ps{ps: 可以拓展到更多的空间维度, e.g. \scalebox{0.9}{$G(x,y,t;\xi,\eta,\tau)$ $\to$ $L G(x,y,t;\xi,\eta,\tau) = \delta(x-\xi)\delta(y-\eta)\delta(t-\tau)$}}

        - How to solve: $L G(x,t;\xi,\tau) = \delta(x-\xi)\delta(t-\tau)$ with BCs and ICs. $\downarrow$

        \quad \ok{a. Fourier Transform} (infinite space domain ($x,t$), \ps{[2 dimensions]})

        \vspace{-2pt}
        \begin{enumerate}
            \item Take Fourier transform in \imp{$x$} $\to$ Get ODE $\to$ Solve ODE with \ok{$t\ne\tau$}. $\to$ Get $\overline{G}$ split by $t<\tau$ and $t>\tau$.
            \vspace{-2pt}
            \item ICs $\to$ Get some unknown constants in $\overline{G}$.
            \vspace{-2pt}
            \item Apply \ok{Jump Condition} to \textit{ODE}, $\int_{\tau-}^{\tau+} \cdot \ dt$ $\to$ Get unknown constant.
            \vspace{-2pt}
            \item Inverse Fourier $\to$ $G(x,t;\xi,\tau)$.
        \end{enumerate}

        \quad \ok{b. Separation of Variables} (finite space domain ($x$))
        
        \vspace{-2pt}
        \begin{enumerate}
            \item Need homogeneous BCs $\to$ Assume \ok{$t\ne \tau$ and $x\ne \xi$} $\to$ Seperation of variables $\to$ Get \ps{$\lambda_n$ and $X_n(x)$.}
            \vspace{-2pt}
            \item Assume $G = \sum_n$ \ok{$g_n(t)$} $X_n(x)$ $\to$ Substitute into PDE $L G=\delta(x-\xi)\delta(t-\tau)$.
            \vspace{-2pt}
            \item Use orthogonality of $X_n$ $\to$ Get $\delta(x-\xi) = \sum_n d_n X_n(x)$ and find $d_n$.
            \vspace{-2pt}
            \item 通过前面substitute得到的对比terms $\to$ Get ODE for $g_n(t)$ $\to$ Solve ODE split by $t<\tau$ and $t>\tau$.
            \vspace{-2pt}
            \item ICs $\to$ Get some unknown constants in $g_n(t)$. $\to$ \ok{Continuity} at $t=\tau$ $\to$ some unknown constants.
            \vspace{-2pt}
            \item Apply \ok{Jump Condition} to \textit{ODE}, $\int_{\tau-}^{\tau+} \cdot \ dt$ $\to$ Get unknown constant $\to$ Get $g_n(t)$ and we solve it :)
        \end{enumerate}

        \qquad \ps{Remark}: \scalebox{0.9}{可以拓展为更多空间维度 $\to$ Separation of variables 后留空间不算到最后假设来找.}

        \qquad \scalebox{0.9}{\ps{e.g. $G_t-\Delta G = \delta(x-\xi)\delta(y-\eta)\delta(t-\tau)$ $\to \cdots \to$ $G=\sum_{m,n} g_{m,n}(t)X_n(x)Y_m(y)$ $\to \cdots$}}

        \quad \ok{c. High Variables / Laplace Transform}
    \end{MyTh}

    \begin{MyTh}{Remark} \tiny
        如$F(x,t)$ 给具体函数 $\to$ 考虑\ok{齐次} PDE $\to$ Eigenfunction $X_n(x)$ $\to$ $u=\sum_n g_n(t)X_n(x)$
        
        \quad $\to$ RHS 含$x$的写为$\sum_n d_n X_n(x)$ $\to$ 比较系数 $\to$ 得到 $g_n(t)$ 的ODE $\to$ Solve ODE $\to$ 最后得到 $u(x,t)$

        \quad \ps{i.e. 跳过求$G$的步骤, 直接用$X_n$展开RHS}
    \end{MyTh}
