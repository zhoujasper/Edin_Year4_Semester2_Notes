    \section{Missing Data Pattern \& Mechanism}

    \begin{MyTh}{Mechanism}
        Insight about \ok{reasons for missingness}. \quad \ps{$\mathbf{Y}=(\mathbf{Y}_{\text{obs},}\mathbf{Y}_{\text{mis}})$ \ \ps{$\mathbf{R}$: \tiny 缺失指标矩阵}}

        \begin{enumerate}
            \item \ok{MCAR}: \ok{a.} {\tiny 缺失数据与已/未观测到变量无关}. \ \ \ok{b.} {\tiny 观察到的数据可以作为完整数据. ($\mathbf{Y}_{\text{obs}}\sim \mathbf{Y}_{\text{mis}} \sim Y$)} \\
            \ok{c.} {\tiny 所有对\textit{完整数据分析}的方法均对MCAR假设下的观察数据有效.} \ \ \ok{d.} {\tiny \ok{$f(\mathbf{r}|\mathbf{y},\psi)=f(\mathbf{r}|\psi)$}.}
            \item \ok{MAR}: \ok{a.} {\tiny 缺失数据\textit{仅与观测到的变量}有关,与其本身无关.} \ \ \ok{b.} {\tiny \ps{把观测变量分层starata;在同一层中,缺失数据的概率因是随机的(MCAR)}. (e.g.用t-test检测分组平均值)} \ \ \ok{c.} {\tiny \ok{$f(\mathbf{r}|\mathbf{y},\psi)=f(\mathbf{r}|\mathbf{y}_{\text{obs}},\psi)$}.}
            \item \ok{MNAR}: \ok{a.} {\tiny 趋势数据\textit{与本身数值}有关.} \quad \ok{b.} {\tiny 或已知缺失与一未观测变量(无法观测)有关.}
        \end{enumerate}
        
        \qquad \ \ps{\tiny NB: Any dataset can possibly be MNAR.} \qquad \ok{c.} \ok{\tiny $f(\mathbf{r}|\mathbf{y},\psi)=f(\mathbf{r}|\mathbf{y}_{\text{obs}},\mathbf{y}_{\text{mis}},\psi)$ .}

        \qquad \ \ps{\tiny ps: To prevent MNAR $\rightarrow$ \ok{Inclusive analysis strategy}: \scalebox{0.9}{数据收集时多收集可能与缺失有关的变量}(auxiliary var.).}
    \end{MyTh}

    \begin{MyTh}{Pattern}
        Insight about \ok{localtions of the missing values} in a dataset. \quad \ps{\tiny NB: Not about reasons!}
        
        \begin{enumerate}
            \item \ok{Univariate}: {\tiny 只有一种变量存在缺失数据} \qquad\qquad\quad \scalebox{0.9}{\tiny 常见于longitudinal study $\downarrow$}
            \item \ok{Monotone}: {\tiny 对变量进行排序,前一个变量的数据缺失会有后续变量的数据都缺失}.
            \item \ok{Arbitrary/General}: {\tiny 任意一个变量中的数据都有可能缺失}.
        \end{enumerate}
    \end{MyTh}

    \begin{MyTh}{Notation}\tiny
        \ok{$\theta$}: param. of \ok{data model} $f(\mathbf{y}|\theta)$ \quad \ok{$\psi$}: param. of \ok{missing data mechanism} $f(\mathbf{r}|\mathbf{y},\psi)$
    \end{MyTh}

    \begin{MyTh}{Rubin}
        The \ps{\textit{Likelihood based inference}} or \ps{\textit{multiple imputation}} don't need $\psi$ if:
        \begin{enumerate}
            \item The data are \ps{MAR} or \ps{MCAR}. \qquad 2. \ The $\theta$ and $\psi$ are \ps{distinct}. ($(\theta,\psi)\in\Theta\times\Psi$)
        \end{enumerate}
    \end{MyTh}
    }}
